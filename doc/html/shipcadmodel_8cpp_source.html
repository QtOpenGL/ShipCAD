177 void ShipCADModel::loadBinary(FileBuffer& source)
178 {
179  // remember the filename because it is erased by the clear method
180  QString tmpstr = _filename;
181  clear();
182  _filename = tmpstr;
183  source.reset();
184  QString hdr;
185  source.load(hdr);
186  if (hdr == "FREE!ship") {
187  int ver, cver;
188  source.load(ver);
189  _file_version = versionFromInt(ver);
190  cver = versionToInt(k_current_version);
191  source.setVersion(_file_version);
192  if (_file_version <= cver) {
193  int n;
194  source.load(n);
195  _precision = static_cast<precision_t>(n);
196  _vis.loadBinary(source);
197  _settings.loadBinary(source, 0);
198  // load the subdivision surface
199  _surface->loadBinary(source);
200  // stations
201  source.load(n);
202  for (int i=0; i<n; i++) {
203  Intersection* intersection = new Intersection(this);
204  intersection->loadBinary(source);
205  _stations.add(intersection);
206  }
207  // buttocks
208  source.load(n);
209  for (int i=0; i<n; i++) {
210  Intersection* intersection = new Intersection(this);
211  intersection->loadBinary(source);
212  _buttocks.add(intersection);
213  }
214  // waterlines
215  source.load(n);
216  for (int i=0; i<n; i++) {
217  Intersection* intersection = new Intersection(this);
218  intersection->loadBinary(source);
219  _waterlines.add(intersection);
220  }
221  if (_file_version >= fv180) {
222  // diagonals
223  source.load(n);
224  for (int i=0; i<n; i++) {
225  Intersection* intersection = new Intersection(this);
226  intersection->loadBinary(source);
227  _diagonals.add(intersection);
228  }
229  if (_file_version >= fv191) {
230  // markers
231  Marker* marker = new Marker(this);
232  marker->loadBinary(source);
233  _markers.add(marker);
234  if (_file_version >= fv210) {
235  // resistance and Kaper
236  }
237  }
238  }
239  }
240  else {
241  // TODO version is later than this can handle
242  }
243  }
244  else {
245  // TODO this is not a free ship binary file
246  }
247  _file_changed = false;
248  _surface->setDesiredSubdivisionLevel(static_cast<int>(_precision)+1);
249  _surface->rebuild();
250  emit onUpdateGeometryInfo();
251 }

253 float ShipCADModel::findLowestHydrostaticsPoint()
254 {
255  float result = _surface->getMin().z();
256  bool first = true;
257  for (size_t i=0; i<numberOfLayers(); i++) {
258  SubdivisionLayer* layer = getLayer(i);
259  if (layer->useInHydrostatics()) {
260  for (size_t j=0; j<layer->numberOfFaces(); j++) {
261  if (first) {
262  result = layer->getFace(j)->getMin().z();
263  first = false;
264  } else if (layer->getFace(j)->getMin().z() < result) {
265  result = layer->getFace(j)->getMin().z();
266  }
267  }
268  }
269  }
270  return result;
271 }
