370 void ShipCADModel::loadBinary(FileBuffer& source)
371 {
372  // remember the filename because it is erased by the clear method
373  QString tmpstr = _filename;
374  clear();
375  _filename = tmpstr;
376  source.reset();
377  QString hdr;
378  source.load(hdr);
379  if (hdr == "FREE!ship") {
380  quint8 v, cver;
381  source.load(v);
382  _file_version = static_cast<version_t>(v);
383  cver = static_cast<quint8>(k_current_version);
384  source.setVersion(_file_version);
385  if (_file_version <= cver) {
386  quint32 n;
387  source.load(n);
388  _precision = static_cast<precision_t>(n);
389  _vis.loadBinary(source);
390  _settings.loadBinary(source, 0);
391  // load the subdivision surface
392  _surface.loadBinary(source);
393  // stations
394  source.load(n);
395  for (size_t i=0; i<n; i++) {
396  Intersection* intersection = new Intersection(this);
397  intersection->loadBinary(source);
398  _stations.add(intersection);
399  }
400  // buttocks
401  source.load(n);
402  for (size_t i=0; i<n; i++) {
403  Intersection* intersection = new Intersection(this);
404  intersection->loadBinary(source);
405  _buttocks.add(intersection);
406  }
407  // waterlines
408  source.load(n);
409  for (size_t i=0; i<n; i++) {
410  Intersection* intersection = new Intersection(this);
411  intersection->loadBinary(source);
412  _waterlines.add(intersection);
413  }
414  if (_file_version >= fv180) {
415  // diagonals
416  source.load(n);
417  for (size_t i=0; i<n; i++) {
418  Intersection* intersection = new Intersection(this);
419  intersection->loadBinary(source);
420  _diagonals.add(intersection);
421  }
422  if (_file_version >= fv191) {
423  // markers
424  source.load(n);
425  for (size_t i=0; i<n; i++) {
426  Marker* marker = new Marker(this);
427  marker->loadBinary(source);
428  _markers.add(marker);
429  }
430  if (_file_version >= fv210) {
431  source.load(&_delft_resistance);
432  source.load(&_kaper_resistance);
433  if (_file_version >= fv250) {
434  source.load(n);
435  for (size_t i=0; i<n; i++) {
436  BackgroundImage* img = new BackgroundImage(this);
437  _background_images.add(img);
438  img->loadBinary(source);
439  }
440  source.load(n);
441  for (size_t i=0; i<n; i++) {
442  Flowline* flow = Flowline::construct(this);
443  _flowlines.add(flow);
444  flow->loadBinary(source);
445  }
446  }
447  }
448  }
449  }
450  }
451  else {
452  // TODO version is later than this can handle
453  }
454  }
455  else {
456  // TODO this is not a free ship binary file
457  }
458  _file_changed = false;
459  rebuildModel();
460  emit onUpdateGeometryInfo();
461 }

509 float ShipCADModel::findLowestHydrostaticsPoint()
510 {
511  float result = _surface.getMin().z();
512  bool first = true;
513  for (size_t i=0; i<numberOfLayers(); i++) {
514  SubdivisionLayer* layer = getLayer(i);
515  if (layer->useInHydrostatics()) {
516  for (size_t j=0; j<layer->numberOfFaces(); j++) {
517  if (first) {
518  result = layer->getFace(j)->getMin().z();
519  first = false;
520  } else if (layer->getFace(j)->getMin().z() < result) {
521  result = layer->getFace(j)->getMin().z();
522  }
523  }
524  }
525  }
526  return result;
527 }
