225 void ShipCADModel::loadBinary(FileBuffer& source)
226 {
227  // remember the filename because it is erased by the clear method
228  QString tmpstr = _filename;
229  clear();
230  _filename = tmpstr;
231  source.reset();
232  QString hdr;
233  source.load(hdr);
234  if (hdr == "FREE!ship") {
235  quint8 v, cver;
236  source.load(v);
237  _file_version = static_cast<version_t>(v);
238  cver = static_cast<quint8>(k_current_version);
239  source.setVersion(_file_version);
240  if (_file_version <= cver) {
241  quint32 n;
242  source.load(n);
243  _precision = static_cast<precision_t>(n);
244  _vis.loadBinary(source);
245  _settings.loadBinary(source, 0);
246  // load the subdivision surface
247  _surface.loadBinary(source);
248  // stations
249  source.load(n);
250  for (size_t i=0; i<n; i++) {
251  Intersection* intersection = new Intersection(this);
252  intersection->loadBinary(source);
253  _stations.add(intersection);
254  }
255  // buttocks
256  source.load(n);
257  for (size_t i=0; i<n; i++) {
258  Intersection* intersection = new Intersection(this);
259  intersection->loadBinary(source);
260  _buttocks.add(intersection);
261  }
262  // waterlines
263  source.load(n);
264  for (size_t i=0; i<n; i++) {
265  Intersection* intersection = new Intersection(this);
266  intersection->loadBinary(source);
267  _waterlines.add(intersection);
268  }
269  if (_file_version >= fv180) {
270  // diagonals
271  source.load(n);
272  for (size_t i=0; i<n; i++) {
273  Intersection* intersection = new Intersection(this);
274  intersection->loadBinary(source);
275  _diagonals.add(intersection);
276  }
277  if (_file_version >= fv191) {
278  // markers
279  source.load(n);
280  for (size_t i=0; i<n; i++) {
281  Marker* marker = new Marker(this);
282  marker->loadBinary(source);
283  _markers.add(marker);
284  }
285  if (_file_version >= fv210) {
286  source.load(&_delft_resistance);
287  source.load(&_kaper_resistance);
288  if (_file_version >= fv250) {
289  source.load(n);
290  for (size_t i=0; i<n; i++) {
291  BackgroundImage* img = new BackgroundImage(this);
292  _background_images.add(img);
293  img->loadBinary(source);
294  }
295  source.load(n);
296  for (size_t i=0; i<n; i++) {
297  Flowline* flow = Flowline::construct(this);
298  _flowlines.add(flow);
299  flow->loadBinary(source);
300  }
301  }
302  }
303  }
304  }
305  }
306  else {
307  // TODO version is later than this can handle
308  }
309  }
310  else {
311  // TODO this is not a free ship binary file
312  }
313  _file_changed = false;
314  _surface.setDesiredSubdivisionLevel(static_cast<int>(_precision)+1);
315  _surface.rebuild();
316  emit onUpdateGeometryInfo();
317 }

319 void ShipCADModel::saveBinary(FileBuffer& dest)
320 {
321  dest.add("FREE!ship");
322  dest.add(static_cast<quint8>(_file_version));
323  dest.add(static_cast<quint32>(_precision));
324  _vis.saveBinary(dest);

326  _surface.saveBinary(dest);
327  // save stations
328  dest.add(_stations.size());
329  for (size_t i=0; i<_stations.size(); i++)
330  _stations.get(i)->saveBinary(dest);
331  // save buttocks
332  dest.add(_buttocks.size());
333  for (size_t i=0; i<_buttocks.size(); i++)
334  _buttocks.get(i)->saveBinary(dest);
335  // save waterlines
336  dest.add(_waterlines.size());
337  for (size_t i=0; i<_waterlines.size(); i++)
338  _waterlines.get(i)->saveBinary(dest);
339  if (getFileVersion() >= fv180) {
340  // save diagonals
341  dest.add(_diagonals.size());
342  for (size_t i=0; i<_diagonals.size(); i++)
343  _diagonals.get(i)->saveBinary(dest);
344  if (getFileVersion() >= fv191) {
345  // save markers
346  dest.add(_markers.size());
347  for (size_t i=0; i<_markers.size(); i++)
348  _markers.get(i)->saveBinary(dest);
349  if (getFileVersion() >= fv210) {
350  dest.add(&_delft_resistance);
351  dest.add(&_kaper_resistance);
352  if (getFileVersion() >= fv250) {
353  dest.add(_background_images.size());
354  for (size_t i=0; i<_background_images.size(); i++)
355  _background_images.get(i)->saveBinary(dest);
356  dest.add(_flowlines.size());
357  for (size_t i=0; i<_flowlines.size(); i++)
358  _flowlines.get(i)->saveBinary(dest);

365 float ShipCADModel::findLowestHydrostaticsPoint()
366 {
367  float result = _surface.getMin().z();
368  bool first = true;
369  for (size_t i=0; i<numberOfLayers(); i++) {
370  SubdivisionLayer* layer = getLayer(i);
371  if (layer->useInHydrostatics()) {
372  for (size_t j=0; j<layer->numberOfFaces(); j++) {
373  if (first) {
374  result = layer->getFace(j)->getMin().z();
375  first = false;
376  } else if (layer->getFace(j)->getMin().z() < result) {
377  result = layer->getFace(j)->getMin().z();
378  }
379  }
380  }
381  }
382  return result;
383 }
