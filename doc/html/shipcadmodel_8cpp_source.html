137 void ShipCADModel::loadBinary(FileBuffer& source)
138 {
139  // remember the filename because it is erased by the clear method
140  QString tmpstr = _filename;
141  clear();
142  _filename = tmpstr;
143  source.reset();
144  QString hdr;
145  source.load(hdr);
146  if (hdr == "FREE!ship") {
147  int ver, cver;
148  source.load(ver);
149  _file_version = versionFromInt(ver);
150  cver = versionToInt(k_current_version);
151  source.setVersion(_file_version);
152  if (_file_version <= cver) {
153  int n;
154  source.load(n);
155  _precision = static_cast<precision_t>(n);
156  _vis.loadBinary(source);
157  _settings.loadBinary(source, 0);
158  // load the subdivision surface
159  _surface->loadBinary(source);
160  // stations
161  source.load(n);
162  for (int i=0; i<n; i++) {
163  Intersection* intersection = Intersection::construct(this);
164  intersection->loadBinary(source);
165  _stations.add(intersection);
166  }
167  // buttocks
168  source.load(n);
169  for (int i=0; i<n; i++) {
170  Intersection* intersection = Intersection::construct(this);
171  intersection->loadBinary(source);
172  _buttocks.add(intersection);
173  }
174  // waterlines
175  source.load(n);
176  for (int i=0; i<n; i++) {
177  Intersection* intersection = Intersection::construct(this);
178  intersection->loadBinary(source);
179  _waterlines.add(intersection);
180  }
181  if (_file_version >= fv180) {
182  // diagonals
183  source.load(n);
184  for (int i=0; i<n; i++) {
185  Intersection* intersection = Intersection::construct(this);
186  intersection->loadBinary(source);
187  _diagonals.add(intersection);
188  }
189  if (_file_version >= fv191) {
190  // markers
191  Marker* marker = new Marker(this);
192  marker->loadBinary(source);
193  _markers.add(marker);
194  if (_file_version >= fv210) {
195  // resistance and Kaper
196  }
197  }
198  }
199  }
200  else {
201  // TODO version is later than this can handle
202  }
203  }
204  else {
205  // TODO this is not a free ship binary file
206  }
207  _file_changed = false;
208  _surface->setDesiredSubdivisionLevel(static_cast<int>(_precision)+1);
209  _surface->rebuild();
210  emit onUpdateGeometryInfo();
211 }
