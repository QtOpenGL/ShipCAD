94 SubdivisionControlPoint* SubdivisionSurface::addControlPoint(const QVector3D& pt)
95 {
96  SubdivisionControlPoint* result = 0;
97  double max_error = 1E-5;
98  for (size_t i=1; i<=numberOfControlEdges(); ++i) {
99  SubdivisionControlEdge* edge = getControlEdge(i-1);
100  if (edge->numberOfFaces() <= 1) { // boundary edge
101  if (SquaredDistPP(pt, edge->startPoint()->getCoordinate()) <= max_error) {
102  result = dynamic_cast<SubdivisionControlPoint*>(edge->startPoint());
103  break;
104  }
105  else if (SquaredDistPP(pt, edge->endPoint()->getCoordinate()) <= max_error) {
106  result = dynamic_cast<SubdivisionControlPoint*>(edge->endPoint());
107  break;
108  }
109  }
110  }
111  if (result == 0) {
112  // search controlpoints without edges
113  for (size_t i=1; i<=numberOfControlPoints(); ++i) {
114  SubdivisionControlPoint* point = getControlPoint(i-1);
115  if (point->numberOfEdges() == 0) {
116  if (SquaredDistPP(pt, point->getCoordinate()) <= max_error) {
117  result = point;
118  break;
119  }
120  }
121  }
122  }
123  if (result == 0) {
124  result = newControlPoint(pt);
125  }
126  return result;
127 }

624 SubdivisionControlFace* SubdivisionSurface::addControlFace(std::vector<QVector3D>& points)
625 {
626  SubdivisionControlFace* result = 0;
627  // remove double points
628  size_t i = 0;
629  double max_error = 1 / pow(10.0, sDecimals);
630  while (i < points.size() - 1) {
631  size_t j = i + 1;
632  while (j < points.size()) {
633  float dist = QVector3D(points[i] - points[j]).length();
634  if (dist <= max_error) {
635  points.erase(points.begin()+j);
636  }
637  else
638  ++j;
639  }
640  ++i;
641  }
642  SubdivisionPoint* point;
643  SubdivisionControlEdge* edge;
644  SubdivisionPoint* prev = 0;
645  if (points.size() > 2) {
646  result = SubdivisionControlFace::construct(this);
647  for (size_t i=1; i<=points.size(); ++i) {
648  point = addControlPoint(points[i-1]);
649  result->addPoint(point);
650  if (i > 1) {
651  edge = addControlEdge(prev, point);
652  edge->addFace(result);
653  }
654  prev = point;
655  }
656  point = result->getPoint(0);
657  edge = addControlEdge(prev, point);
658  edge->addFace(result);
659 
660  // check if a point refers to the same face more than once ==> invalid face
661  bool invalidface = false;
662  for (size_t i=1; i<=result->numberOfPoints(); ++i) {
663  size_t n = 0;
664  point = result->getPoint(i-1);
665  for (size_t j=1; j<=point->numberOfFaces(); ++j) {
666  if (point->getFace(j-1) == result)
667  n++;
668  }
669  if (n > 1) {
670  invalidface = true;
671  break;
672  }
