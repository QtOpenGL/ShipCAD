81 static float triangle_area(const QVector3D& p1, const QVector3D& p2, const QVector3D& p3)
82 {
83  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z())
84  + (p2.y() - p3.y()) * (p2.z() + p3.z())
85  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
86  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x())
87  + (p2.z() - p3.z()) * (p2.x() + p3.x())
88  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
89  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y())
90  + (p2.x() - p3.x()) * (p2.y() + p3.y())
91  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
92  return sqrt(ax*ax + ay*ay + az*az);
93 }

118 QVector3D SubdivisionFace::getFaceNormal()
119 {
120  QVector3D result = ZERO;
121  QVector3D c = ZERO;
122 
123  if (_points.size() == 0)
124  return result;
125 
126  // calculate center of the face
127  for (size_t i=0; i<_points.size(); ++i) {
128  QVector3D p = _points[i]->getCoordinate();
129  c += p;
130  }
131  c /= _points.size();
132  // calculate normal
133  SubdivisionPoint* p1 = _points[_points.size() - 1];
134  for (size_t i=0; i<_points.size(); ++i) {
135  SubdivisionPoint* p2 = _points[i];
136  QVector3D n = UnifiedNormal(c, p1->getCoordinate(), p2->getCoordinate());
137  result += n;
138  p1 = p2;
139  }
140  result.normalize();
141  return result;

259 void SubdivisionFace::subdivide(bool controlface,
260  vector<pair<SubdivisionPoint*,SubdivisionPoint*> > &vertexpoints,
261  vector<pair<SubdivisionEdge*,SubdivisionPoint*> > &edgepoints,
262  vector<pair<SubdivisionFace*,SubdivisionPoint*> > &facepoints,
263  vector<SubdivisionEdge*> &interioredges,
264  vector<SubdivisionEdge*> &controledges,
265  vector<SubdivisionFace*> &dest)
266 {
267  SubdivisionPoint* pts[4];
268  SubdivisionFace* newface;
269  SubdivisionPoint* p2;
270  SubdivisionEdge* prevedge;
271  SubdivisionEdge* curedge;
272  vector<pair<SubdivisionPoint*,SubdivisionPoint*> >::iterator ptmpindex;
273  vector<pair<SubdivisionEdge*,SubdivisionPoint*> >::iterator etmpindex;
274  vector<pair<SubdivisionFace*,SubdivisionPoint*> >::iterator ftmpindex;
275 
276  if (_points.size() != 3 || _owner->getSubdivisionMode() == fmCatmullClark) {
277  for (size_t i=1; i<=_points.size(); ++i) {
278  p2 = _points[i-1];
279  size_t index = (i + _points.size() - 2) % _points.size();
280  prevedge = _owner->edgeExists(p2, _points[index]);
281  index = (i + _points.size()) % _points.size();
282  curedge = _owner->edgeExists(p2, _points[index]);
283  index = (i - 1) % 4;
284  ptmpindex = find_if(vertexpoints.begin(), vertexpoints.end(), PointPred(p2));
285  pts[index] = (*ptmpindex).second; // p2.newlocation
286  SubdivisionPoint* p2pt = pts[index];
287  index = (index + 1) % 4;
288  etmpindex = find_if(edgepoints.begin(), edgepoints.end(), EdgePred(curedge));
289  pts[index] = (*etmpindex).second; // curedge.newlocation
290  SubdivisionPoint* curredgept = pts[index];
291  index = (index + 1) % 4;
292  ftmpindex = find_if(facepoints.begin(), facepoints.end(), FacePred(this));
293  pts[index] = (*ftmpindex).second; // this.newlocation
294  SubdivisionPoint* newlocation = pts[index];
295  index = (index + 1) % 4;
296  etmpindex = find_if(edgepoints.begin(), edgepoints.end(), EdgePred(prevedge));
297  pts[index] = (*etmpindex).second; // prevedge.newlocation
298  SubdivisionPoint* prevedgept = pts[index];
299  // add the new face
300  newface = SubdivisionFace::construct(_owner);
301  dest.push_back(newface);
302  // check the edges of the new face
303  newface->edgeCheck(prevedgept, p2pt, prevedge->isCrease(),
304  prevedge->isControlEdge() || controlface,
305  prevedge->getCurve(), interioredges, controledges);
306  newface->edgeCheck(p2pt, curredgept, curedge->isCrease(),
307  curedge->isControlEdge() || controlface,
308  curedge->getCurve(), interioredges, controledges);
309  newface->edgeCheck(curredgept, newlocation, false, false, 0,
310  interioredges, controledges);
311  newface->edgeCheck(prevedgept, newlocation, false, false, 0,
312  interioredges, controledges);
313  for (size_t j=0; j<4; ++j) {
314  // add new face to points
315  pts[j]->addFace(newface);
316  newface->addPoint(pts[j]);
317  }
318  }
319  }
320  else if (numberOfPoints() == 3) {
321  // special case, quadrisect triangle by connecting all three
322  // edge points
323  // first the three surrounding triangles
324  for (size_t i=1; i<=_points.size(); ++i) {
325  p2 = _points[i-1];
326  size_t index = (i-2+numberOfPoints()) % numberOfPoints();
327  prevedge = _owner->edgeExists(p2, _points[index]);
328  index = (i+numberOfPoints()) % numberOfPoints();
329  curedge = _owner->edgeExists(p2, _points[index]);
330 
331  index = 0;
332  etmpindex = find_if(edgepoints.begin(), edgepoints.end(), EdgePred(prevedge));
333  pts[index] = (*etmpindex).second;
334  index = 1;
335  ptmpindex = find_if(vertexpoints.begin(), vertexpoints.end(), PointPred(p2));
336  pts[index] = (*ptmpindex).second;
337  index = 2;
338  etmpindex = find_if(edgepoints.begin(), edgepoints.end(), EdgePred(curedge));
339  pts[index] = (*etmpindex).second;
340  // add the new face
341  newface = SubdivisionFace::construct(_owner);
342  dest.push_back(newface);
343  // check the edges of the new face
344  newface->edgeCheck(pts[0], pts[1], prevedge->isCrease(),
345  prevedge->isControlEdge() || controlface,
346  prevedge->getCurve(), interioredges, controledges);
347  newface->edgeCheck(pts[1], pts[2], curedge->isCrease(),
348  curedge->isControlEdge() || controlface,
349  curedge->getCurve(), interioredges, controledges);
350  newface->edgeCheck(pts[2], pts[0], false, false, 0,
351  interioredges, controledges);
352  for (size_t j=0; j<3; ++j) {
353  // add new face to points
354  pts[j]->addFace(newface);
355  newface->addPoint(pts[j]);
356  }
357  }
358 
359  // then the center triangle
360  for (size_t i=1; i<=numberOfPoints(); ++i) {
361  p2 = _points[i-1];
362  size_t index = (i - 2 + numberOfPoints()) % numberOfPoints();
363  prevedge = _owner->edgeExists(p2, _points[index]);
364  etmpindex = find_if(edgepoints.begin(), edgepoints.end(), EdgePred(prevedge));
365  pts[i-1] = (*etmpindex).second;
366  }
367  // add the new face
368  newface = SubdivisionFace::construct(_owner);
369  dest.push_back(newface);
370  newface->edgeCheck(pts[0], pts[1], false, false, 0,
371  interioredges, controledges);
372  newface->edgeCheck(pts[1], pts[2], false, false, 0,
373  interioredges, controledges);
374  newface->edgeCheck(pts[2], pts[0], false, false, 0,
375  interioredges, controledges);
376  for (size_t j=0; j<3; ++j) {
377  // add new face to points
378  pts[j]->addFace(newface);
379  newface->addPoint(pts[j]);
380  }
381  }

593 SubdivisionControlEdge* SubdivisionControlFace::insertControlEdge(
594  SubdivisionControlPoint *p1, SubdivisionControlPoint *p2)
595 {
596  SubdivisionControlEdge* result = 0;
597  if (p1->hasFace(this) && p2->hasFace(this)) {
598  if (_owner->edgeExists(p1, p2) != 0)
599  return result;
600  size_t tmp = indexOfPoint(p1);
601  vector<SubdivisionControlPoint*> pts;
602  pts.push_back(p1);
603  for (size_t i=0; i<numberOfPoints(); ++i) {
604  tmp = (tmp + 1) % numberOfPoints();
605  pts.push_back(dynamic_cast<SubdivisionControlPoint*>(getPoint(tmp)));
606  if (pts[pts.size()-1] == p2)
607  break;
608  }

700 void SubdivisionControlFace::calcExtents()
701 {
702  // calculate min/max coordinate of all children
703  if (numberOfPoints() > 0)
704  _min = getPoint(0)->getCoordinate();
705  else
706  _min = ZERO;
707  _max = _min;
708  if (_children.size() > 0) {
709  for (size_t i=1; i<=_children.size(); ++i) {
710  SubdivisionFace* face = _children[i-1];
711  for (size_t j=1; j<=face->numberOfPoints(); ++j) {
712  if (i == 1 && j == 1) {
713  _min = face->getPoint(j - 1)->getCoordinate();
714  _max = _min;
715  }
716  MinMax(face->getPoint(j - 1)->getCoordinate(), _min, _max);
717  }
718  }
719  }
720  else {
721  for (size_t i=2; i<=numberOfPoints(); ++i) {
722  MinMax(getPoint(i - 1)->getCoordinate(), _min, _max);
723  }
724  }
