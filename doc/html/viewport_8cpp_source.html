153 // this procedure initializes the viewport and sets the scale in such a way
154 // that the model completely fills the viewport
155 void Viewport::initializeViewport(const QVector3D& min, const QVector3D& max)
156 {
157  // calculate the midpoint of the bounding box, which is used as the center of the
158  // model for rotating the model
159  _midpoint = 0.5 * (_min3d + _max3d);
160  // calculate the distance of the camera to the center of the model, following from
161  // the field of view from the camera
162  float dist = sqrt((_max3d.y() - _min3d.y()) * (_max3d.y() - _min3d.y())
163  + (_max3d.z() - _min3d.z()) * (_max3d.z() - _min3d.z()));
164  if (dist == 0)
165  dist = 1E-2f;
166  if (_view_type == fvPerspective) {
167  if (atan(_field_of_view) != 0) {
168  _distance = 1.5 * dist / atan(_field_of_view);
169  if (_distance > 1E5)
170  _distance = 1E5;
171  }
172  else
173  _distance = 1E5;
174  }
175  else
176  _distance = 1E8;
177 
178  // build the vertex transformation matrix from the perspective
179  // and the angle, elevation
180 
181  _proj = QMatrix4x4();
182  _proj.perspective(RadToDeg(_field_of_view), width() / static_cast<float>(height()), 0.1f, 100.0f);
183 
184  QMatrix4x4 model;
185  // find the camera location
186  model.rotate(_elevation, 1, 0, 0);
187  model.rotate(_angle, 0, 0, 1);
188  _camera_location = model.map(QVector3D(_max3d.x() + _distance, 0, 0));
189 
190  // perspective view matrix
191  QMatrix4x4 view;
192  view.lookAt(_camera_location, _midpoint, QVector3D(0,0,1));
193  _view = view;
194 
195  // final matrix
196  _matrix = _proj * _view;
197 
198  cout << "angle: " << _angle << " elev: " << _elevation << endl;
199  renderLater();
200 }

275 void
276 Viewport::convertMouseCoordToWorld(int mx, int my)
277 {
278  float x = mx;
279  float y = my;
280 
281  QVector3D view = _midpoint - _camera_location;
282  view.normalize();
283  QVector3D h = QVector3D::crossProduct(view, QVector3D(0,0,1));
284  h.normalize();
285  QVector3D v = QVector3D::crossProduct(h, view);
286  v.normalize();
287 
288  float vlength = tan(_field_of_view / 2.0f) * 0.1f;
289  float hlength = vlength * width() / height();
290  v *= vlength;
291  h *= hlength;
292 
293  // translate mouse coordinates so that origin lies in the center of the viewport
294  x -= width() / 2.0f;
295  y -= height() / 2.0f;
296 
297  // scale mouse coordinates so that half the viewport width and height becomes 1
298  x /= width() / 2.0f;
299  y /= height() / 2.0f;
300 
301  // linear combination to compute intersection of picking ray with viewport plane
302  QVector3D pos = _camera_location + view * 0.1f + h * x + v * y;
303  QVector3D dir = pos - _camera_location;
304  dir.normalize();
305 
306  cout << "mouse lb release model coord: " << pos.x()
307  << "," << pos.y()
308  << "," << pos.z() << endl;
309  // lets find the distance from the 1,1,0 point
310  QVector3D p1(1,1,0);
311  float dist = p1.distanceToLine(pos, dir);
312  cout << "mouse pos distance to 1,1,0: " << dist << endl;
313 }

336 void
337 Viewport::mouseMoveEvent(QMouseEvent *event)
338 {
339  if (_view_type == fvPerspective && event->buttons().testFlag(Qt::MidButton)) {
340  // dragging the perspective around with middle button
341  _angle -= (event->pos().x() - _prev_pos.x()) / 2.0f;
342  while (_angle > 180)
343  _angle -= 360;
344  while (_angle < -180)
345  _angle += 360;
346  _elevation += -(event->pos().y() - _prev_pos.y()) / 2.0f;
347  while (_elevation > 180)
348  _elevation -= 360;
349  while (_elevation < -180)
350  _elevation += 360;
351  initializeViewport(_min3d, _max3d);
352  }
353  _prev_pos = event->pos();
354  // are we getting mouse clicks in the gl window?
355  cout << "mouse move: " << event->pos().x() << "," << event->pos().y() << endl;
