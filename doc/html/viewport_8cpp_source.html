138 void Viewport::render()
139 {
140 
141  glViewport(0, 0, width(), height());
142 
143  // setup the lighting
144  float ltcolor[] = {0, .3, 1.0};
145  glEnable( GL_LIGHTING );
146  glEnable( GL_LIGHT0 );
147  glLightfv(GL_LIGHT0, GL_POSITION, ltcolor);
148 
149  // set to shade
150  setViewportMode(vmShade);
151 
152  // set the color of the background
153  QColor vpcolor = _ctl->getModel()->getPreferences().getViewportColor();
154  glClearColor(vpcolor.redF(), vpcolor.greenF(), vpcolor.blueF(), 1.0);
155  glClear(GL_COLOR_BUFFER_BIT);
156 
157  // draw the model
158  _ctl->getModel()->draw(*this);
159 
160  // need to release the shader, otherwise doesn't draw right
161  if (_current_shader != 0) {
162  _current_shader->release();
163  _current_shader = 0;
164  }
165 }

229 void
230 Viewport::mouseReleaseEvent(QMouseEvent *event)
231 {
232  bool view_changed = false;
233  // for mouse release, the button released won't be in the set
234  if (_prev_buttons.testFlag(Qt::LeftButton) && !event->buttons().testFlag(Qt::LeftButton)) {
235  if (_drag_state == 1)
236  view_changed = _view->leftMousePick(event->pos(), width(), height());
237  else
238  view_changed = _view->leftMouseRelease(event->pos(), width(), height());
239  }
240  else if (_prev_buttons.testFlag(Qt::RightButton) && !event->buttons().testFlag(Qt::RightButton)) {
241  if (_drag_state == 1)
242  view_changed = contextMenu(event);
243  else
244  view_changed = _view->rightMouseRelease(event->pos(), width(), height());
245  }
246 
247  if (view_changed)
248  update();
249  _prev_buttons = event->buttons();
250  // are we getting mouse clicks in the gl window?
251  cout << "mouse release: " << event->pos().x() << "," << event->pos().y() << endl;
252  _drag_state = 0;
253 }

255 void
256 Viewport::mouseMoveEvent(QMouseEvent *event)
257 {
258  bool view_changed = false;
259 
260  // check for actual dragging
261  if ((_drag_state < 1 || _drag_state > 2)
262  || (_drag_state == 1 && (event->pos() - _drag_start).manhattanLength() < QApplication::startDragDistance()))
263  return;
264 
265  _drag_state = 2;
266 
267  if (event->buttons().testFlag(Qt::LeftButton)) {
268  view_changed = _view->leftMouseMove(event->pos(), _prev_pos, width(), height());
269  }
270  else if (event->buttons().testFlag(Qt::MidButton)) {
271  view_changed = _view->middleMouseMove(event->pos(), _prev_pos, width(), height());
272  }
273  else if (event->buttons().testFlag(Qt::RightButton)) {
274  view_changed = _view->rightMouseMove(event->pos(), _prev_pos, width(), height());
275  }
276 
277  if (view_changed)
278  update();
279 
280  _prev_pos = event->pos();
281  // are we getting mouse clicks in the gl window?
282  //cout << "mouse move: " << event->pos().x() << "," << event->pos().y() << endl;
283 }
