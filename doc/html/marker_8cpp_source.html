63 void Marker::draw(Viewport& vp, LineShader* lineshader)
64 {
65  if (!isVisible())
66  return;
67  if (isSelected())
68  setColor(_owner->getPreferences().getSelectColor());
69  else
70  setColor(_owner->getPreferences().getMarkerColor());
71  setFragments(250);
72  QVector<QVector3D> vertices;
73  QVector3D p3d;
74  if (vp.getViewportType() == fvBodyplan && _owner->getVisibility().getModelView() != mvBoth) {
75 
76  vector<float> params;
77  vector<QVector3D> parray1;
78  vector<QVector3D> parray2;
79 
80  Plane pln(1,0,0,-_owner->getProjectSettings().getMainframeLocation());
81  params.push_back(0);
82  params.push_back(1);
83 
84  IntersectionData output;
85  if (intersect_plane(pln, output)) {
86  params.insert(params.end(), output.parameters.begin(), output.parameters.end());
87  sort(params.begin(), params.end());
88  }
89 
90  for (size_t i=1; i<params.size(); i++) {
91  p3d = value(0.5 * params[i-1] + params[i]);
92  float scale = 1;
93  if (p3d.x() < _owner->getProjectSettings().getMainframeLocation())
94  scale = -1;
95  size_t fragm = floor((params[i] - params[i-1])*getFragments());
96  if (fragm < 10)
97  fragm = 10;
98  for (size_t j=0; j<fragm; j++) {
99  float t = params[i-1] + (params[i] - params[i-1]) * j / (fragm - 1);
100  QVector3D p = value(t);
101  p.setY(p.y() * scale);
102  parray1.push_back(p);
103  }
104  if (showCurvature()) {
105  for (size_t j=0; j<fragm; j++) {
106  float t = params[i-1] + (params[i] - params[i-1]) * j / (fragm - 1);
107  QVector3D normal;
108  float c = curvature(t, normal);
109  normal.setY(normal.y() * scale);
110  QVector3D p2 = parray1[j] - (c * 2 * getCurvatureScale() * normal);
111  parray2.push_back(p2);
112  }
113  for (size_t j=1; j<=fragm; j++) {
114  if ((j % 4) == 0 || j == 1 || j == fragm) {
115  vertices << parray1[j-1];
116  vertices << parray2[j-1];
117  }
118  }
119  for (size_t j=1; j<parray2.size(); j++) {
120  vertices << parray2[j-1];
121  vertices << parray2[j];
122  }
123  glLineWidth(1);
124  lineshader->renderLines(vertices, getCurvatureColor());
125  } else { // end show curvature
126  for (size_t j=1; j<parray1.size(); j++) {
127  vertices << parray1[j-1];
128  vertices << parray1[j];
129  }
130  glLineWidth(1);
131  lineshader->renderLines(vertices, getColor());
132  }
133  }
134  vertices.clear();
135  for (size_t i=0; i<numberOfPoints(); i++) {
136  p3d = getPoint(i);
137  if (p3d.x() < _owner->getProjectSettings().getMainframeLocation())
138  p3d.setY(-p3d.y());
139  vertices << p3d;
140  }
141  glPointSize(3);
142  lineshader->renderPoints(vertices, getColor());
143  } else { // end bodyplan
144  Spline::draw(vp, lineshader);
145  for (size_t i=0; i<numberOfPoints(); i++) {
146  vertices << getPoint(i);
147  }
148  glPointSize(3);
149  lineshader->renderPoints(vertices, getColor());
150  }
151 }
