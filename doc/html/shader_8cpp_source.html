53 void Shader::initialize(const char* vertexShaderSource,
54  const char* fragmentShaderSource,
55  vector<string> uniforms,
56  vector<string> attributes)
57 {
58  _program = new QOpenGLShaderProgram(_viewport);
59  _program->addShaderFromSourceCode(QOpenGLShader::Vertex, vertexShaderSource);
60  _program->addShaderFromSourceCode(QOpenGLShader::Fragment, fragmentShaderSource);
61  if (!_program->link())
62  cerr << "OpenGL Link failed - Log:\n" << _program->log().toStdString() << endl;
63 
64  addUniform("matrix");
65 
66  for (size_t i=0; i<uniforms.size(); ++i)
67  addUniform(uniforms[i]);
68  for (size_t i=0; i<attributes.size(); ++i)
69  addAttribute(attributes[i]);
70 }

202  GLuint vertexAttr = _attributes["vertex"];
203

205  _program->setAttributeArray(normalAttr, normals.constData());
206  _program->enableAttributeArray(normalAttr);
207  _program->enableAttributeArray(vertexAttr);

209  _program->disableAttributeArray(normalAttr);
