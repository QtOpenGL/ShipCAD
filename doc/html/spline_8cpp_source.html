137 void Spline::rebuild()
138 {
139  _build = false;
140 
141  // attempt to eliminate double points
142  _total_length = 0;
143  for (size_t i=1; i<_nopoints; ++i) {
144  _total_length += sqrt(_points[i].distanceToPoint(_points[i-1]));
145  }
146  vector<QVector3D> u;
147  QVector3D un, qn;
148  float sig, p;
149 
150  if (_nopoints > 1) {
151  _derivatives.clear();
152  _derivatives.reserve(_nopoints);
153  _parameters.clear();
154  _parameters.reserve(_nopoints);
155  u.reserve(_nopoints);
156 
157  if (fabs(_total_length) < 1E-5) {
158  for (size_t i=0; i<_nopoints; ++i) {
159  _parameters.push_back(i / static_cast<float>(_nopoints));
160  }
161  }
162  else {
163  _parameters.push_back(0);
164  float length = 0;
165  for (size_t i=1; i<_nopoints-1; ++i) {
166  length += sqrt(_points[i-1].distanceToPoint(_points[i]));
167  _parameters.push_back(length / _total_length);
168  }
169  _parameters.push_back(1.0);
170  }
171 
172  _derivatives.push_back(ZERO);
173  u.push_back(_derivatives[0]);
174 
175  for (size_t i=1; i<_nopoints-1; ++i) {
176  if (_knuckles[i]) {
177  u[i-1] = ZERO;
178  _derivatives.push_back(ZERO);
179  }
180  else {
181  if (fabs(_parameters[i+1] - _parameters[i-1]) < 1e-5
182  || fabs(_parameters[i] - _parameters[i-1]) < 1e-5
183  || fabs(_parameters[i+1] - _parameters[i]) < 1e-5) {
184  _derivatives.push_back(ZERO);
185  }
186  else {
187  sig = (_parameters[i] - _parameters[i-1])
188  / (_parameters[i+1] - _parameters[i-1]);
189  // first x-value
190  p = sig * _derivatives[i-1].x() + 2.0;
191  QVector3D p1;
192  QVector3D p2;
193  p1.setX((sig - 1.0) / p);
194  p2.setX( (6.0 * ((_points[i+1].x() - _points[i].x())
195  / (_parameters[i+1] - _parameters[i])
196  - (_points[i].x() - _points[i-1].x())
197  / (_parameters[i] - _parameters[i-1]))
198  / (_parameters[i+1] - _parameters[i-1])
199  - sig * u[i-1].x()) / p);
200  // then y-value
201  p = sig * _derivatives[i-1].y() + 2.0;
202  p1.setY((sig - 1.0) / p);
203  p2.setY( (6.0 * ((_points[i+1].y() - _points[i].y())
204  / (_parameters[i+1] - _parameters[i])
205  - (_points[i].y() - _points[i-1].y())
206  / (_parameters[i] - _parameters[i-1]))
207  / (_parameters[i+1] - _parameters[i-1])
208  - sig * u[i-1].y()) / p);
209  // then z-value
210  p = sig * _derivatives[i-1].z() + 2.0;
211  p1.setZ((sig - 1.0) / p);
212  p2.setZ( (6.0 * ((_points[i+1].z() - _points[i].z())
213  / (_parameters[i+1] - _parameters[i])
214  - (_points[i].z() - _points[i-1].z())
215  / (_parameters[i] - _parameters[i-1]))
216  / (_parameters[i+1] - _parameters[i-1])
217  - sig * u[i-1].z()) / p);
218  _derivatives.push_back(p1);
219  u.push_back(p2);
220  }
221  }
222  }
223 
224  // BUGBUG: these are always 0, how does it work
225  qn = ZERO;
226  un = qn;
227 
228  QVector3D p3;
229  p3.setX(un.x() - qn.x() * u[_nopoints-2].x()
230  / (qn.x() * _derivatives[_nopoints-2].x() + 1.0));
231  p3.setY(un.y() - qn.y() * u[_nopoints-2].y()
232  / (qn.y() * _derivatives[_nopoints-2].y() + 1.0));
233  p3.setZ(un.z() - qn.z() * u[_nopoints-2].z()
234  / (qn.z() * _derivatives[_nopoints-2].z() + 1.0));
235  _derivatives.push_back(p3);
236 
237  // back substitution
238  for (int k=_nopoints-1; k>=1; --k) {
239  _derivatives[k-1] = _derivatives[k-1] * _derivatives[k] + u[k-1];
240  }
241  } // end _nopoints > 1
242  _build = true;
243  // determine min/max values
244  if (_nopoints > 0) {
245  for (size_t i=0; i<_nopoints; ++i) {
246  if (i == 0) {
247  _min = _points[i];
248  _max = _min;
249  }
250  else {
251  MinMax(_points[i], _min, _max);
252  }
253  }
254  }
255 }

257 QVector3D Spline::second_derive(float parameter)
258 {
259  QVector3D result = ZERO;
260  if (_nopoints < 2)
261  return result;
262  if (!_build)
263  rebuild();
264  if (_nopoints < 2)
265  return result;
266 
267  int lo, hi, k;
268 
269  if (_nopoints == 2) {
270  lo = 0;
271  hi = 1;
272  }
273  else {
274  lo = 0;
275  hi = _nopoints - 1;
276  do {
277  k = (lo + hi) / 2;
278  if (_parameters[k] < parameter)
279  lo = k;
280  else
281  hi = k;
282  } while (hi - lo > 1);
283  }
284  float frac;
285  if (_parameters[hi] - _parameters[lo] <= 0.0)
286  frac = 0.5;
287  else
288  frac = (parameter - _parameters[lo]) / (_parameters[hi] - _parameters[lo]);
289  result = _derivatives[lo] + (frac * (_derivatives[hi] - _derivatives[lo]));
290  return result;
291 }

293 float Spline::weight(size_t index)
294 {
295  float result;
296  float length, dist;
297 
298  if (index == 0 || (index == _nopoints - 1) || _knuckles[index]) {
299  result = 1E10;
300  }
301  else {
302  QVector3D p1 = _points[index-1];
303  QVector3D p2 = _points[index];
304  QVector3D p3 = _points[index+1];
305  length = (p3 - p1).length();
306  if (length < 1E-5) {
307  result = 0.0;
308  }
309  else {
310  dist = DistancepointToLine(p2,p1,p3);
311  if (dist < 1E-2) {
312  if (length*length/_total_length > 0.01)
313  result = 1E10;
314  else
315  result = 1E8 * dist * dist * length;
316  }
317  else
318  result = 1E8 * dist * dist * length;
319  }
320  }
321  return result;
322 }

346 bool Spline::simplify(float criterium)
347 {
348  vector<float> weights;
349  weights.reserve(_nopoints);
350  bool result = false;
351 
352  if (_nopoints < 3)
353  return true;
354 
355  float total_length = _total_length * _total_length;
356  if (total_length == 0)
357  return result;
358 
359  for (size_t i=0; i<_nopoints; ++i)
360  weights.push_back(weight(i)/total_length);
361 
362  vector<float>::iterator index;
363  do {
364  index = find_next_point(weights);
365  vector<float>::iterator last = weights.end();
366  last--;
367  if (index == weights.end() || _nopoints < 3 || index == last)
368  break;
369  if (*index < criterium) {
370  size_t j = index - weights.begin();
371  weights.erase(index);
372  _points.erase(_points.begin()+j);
373  _knuckles.erase(_knuckles.begin()+j);
374  _nopoints--;
375  if (j > 0 && j < _nopoints)
376  weights[j] = weights[j]/total_length;
377  }
378  else
379  break;
380  }
381  while (index != weights.end());
382  result = true;
383 
384  setBuild(false);
385 
386  return result;
387 }

417 float Spline::chord_length_approximation(float percentage)
418 {
419  float result, length, total_length;
420  float t1, t2;
421  t1 = 0;
422  t2 = 1.0;
423  float parameter = 1.0;
424  int counter = 0;
425  float l1 = 0;
426  float l2 = 0;
427  float desired_length = 0;
428  if (percentage < 0)
429  result = 0;
430  else if (percentage > 1)
431  result = 1;
432  else {
433  do {
434  length = coord_length(0, parameter);
435  if (counter == 0) {
436  l1 = 0;
437  l2 = length;
438  total_length = length;
439  desired_length = percentage*total_length;
440  parameter = percentage;
441  }
442  else {
443  if (length > desired_length) {
444  t2 = parameter;
445  l2 = length;
446  }
447  else {
448  t1 = parameter;
449  l1 = length;
450  }
451  parameter = t1 + ((desired_length - l1) / (l2-l1)) * (t2 - t1);
452  if (parameter < 0)
453  parameter = 0;
454  if (parameter > 1)
455  parameter = 1;
456  }
457  counter++;
458  }
459  while (counter <= 75 && fabs(length - desired_length) >= 1E-3);
460  result = parameter;
461  }
462  return result;
463 }

465 float Spline::curvature(float parameter, QVector3D& normal)
466 {
467  float result;
468 
469  QVector3D vel1 = first_derive(parameter);
470  QVector3D acc = second_derive(parameter);
471 
472  QVector3D crossproduct = QVector3D::crossProduct(acc, vel1);
473  float l = crossproduct.length();
474  if (l == 0)
475  l = 0.00001f;
476  float vdota = (vel1.x()*acc.x()) + (vel1.y()*acc.y()) + (vel1.z()*acc.z());
477  float vdotv = (vel1.x()*vel1.x()) + (vel1.y()*vel1.y()) + (vel1.z()*vel1.z());
478  if (vdotv < 0)
479  result = 0;
480  else
481  result = l/pow(vdotv, static_cast<float>(1.5));
482  normal = (vdotv * acc) - (vdota * vel1);
483  normal.normalize();
484  return result;
485 }

617 bool Spline::intersect_plane(const Plane& plane, IntersectionData& output)
618 {
619  output.number_of_intersections = 0;
620  float t1 = 0;
621  QVector3D p1 = value(t1);
622  float s1 = plane.distance(p1);
623  if (fabs(s1) < 1E-6)
624  add_to_output(p1, t1, output);
625  for (size_t i=1; i<=_fragments; ++i) {
626  float t2 = i / static_cast<float>(_fragments);
627  QVector3D p2 = value(t2);
628  float s2 = plane.distance(p2);
629  if (fabs(s2) < 1E-6)
630  add_to_output(p2, t2, output);
631  if ((s1 < 0 && s2 > 0) || (s2 < 0 && s1 > 0)) {
632  // intersection found
633  float t = s1 / (s2 - s1);
634  t = t1 + t * (t2 - t1);
635  add_to_output(value(t), t, output);
636  }
637  p1 = p2;
638  s1 = s2;
639  t1 = t2;
640  }
641  return output.number_of_intersections > 0;
642 }

748 QVector3D Spline::value(float parameter)
749 {
750  QVector3D result;
751  if (_nopoints < 2)
752  return result;
753  if (!_build)
754  rebuild();
755  if (_nopoints < 2)
756  return result;
757 
758  int lo, hi;
759  if (_nopoints == 2) {
760  lo = 0;
761  hi = 1;
762  }
763  else {
764  lo = 0;
765  hi = _nopoints - 1;
766  do {
767  int k = (lo + hi) / 2;
768  if (_parameters[k] < parameter)
769  lo = k;
770  else
771  hi = k;
772  }
773  while (hi - lo > 1);
774  }
775  float h, a, b;
776  h = _parameters[hi] - _parameters[lo];
777  if (fabs(h) < 1E-6) {
778  result = _points[hi];
779  }
780  else {
781  a = (_parameters[hi] - parameter) / h;
782  //b = (parameter - _parameters[lo]) / h;
783  b = 1 - a;
784  result = a * _points[lo] + b * _points[hi] + ((a * a * a - a) * _derivatives[lo]
785  + (b * b * b - b) * _derivatives[hi])
786  * (h * h) / 6.0;
787  }
788  return result;
789 }
