139 void Spline::rebuild()
140 {
141  _build = false;
142 
143  // attempt to eliminate double points
144  _total_length = 0;
145  for (size_t i=1; i<_nopoints; ++i) {
146  _total_length += sqrt(_points[i].distanceToPoint(_points[i-1]));
147  }
148  vector<QVector3D> u;
149  QVector3D un, qn;
150  float sig, p;
151 
152  if (_nopoints > 1) {
153  _derivatives.clear();
154  _derivatives.reserve(_nopoints);
155  _parameters.clear();
156  _parameters.reserve(_nopoints);
157  u.reserve(_nopoints);
158 
159  if (fabs(_total_length) < 1E-5) {
160  for (size_t i=0; i<_nopoints; ++i) {
161  _parameters.push_back(i / static_cast<float>(_nopoints));
162  }
163  }
164  else {
165  _parameters.push_back(0);
166  float length = 0;
167  for (size_t i=1; i<_nopoints-1; ++i) {
168  length += sqrt(_points[i-1].distanceToPoint(_points[i]));
169  _parameters.push_back(length / _total_length);
170  }
171  _parameters.push_back(1.0);
172  }
173 
174  _derivatives.push_back(ZERO);
175  u.push_back(_derivatives[0]);
176 
177  for (size_t i=1; i<_nopoints-1; ++i) {
178  if (_knuckles[i]) {
179  u[i-1] = ZERO;
180  _derivatives.push_back(ZERO);
181  }
182  else {
183  if (fabs(_parameters[i+1] - _parameters[i-1]) < 1e-5
184  || fabs(_parameters[i] - _parameters[i-1]) < 1e-5
185  || fabs(_parameters[i+1] - _parameters[i]) < 1e-5) {
186  _derivatives.push_back(ZERO);
187  }
188  else {
189  sig = (_parameters[i] - _parameters[i-1])
190  / (_parameters[i+1] - _parameters[i-1]);
191  // first x-value
192  p = sig * _derivatives[i-1].x() + 2.0;
193  QVector3D p1;
194  QVector3D p2;
195  p1.setX((sig - 1.0) / p);
196  p2.setX( (6.0 * ((_points[i+1].x() - _points[i].x())
197  / (_parameters[i+1] - _parameters[i])
198  - (_points[i].x() - _points[i-1].x())
199  / (_parameters[i] - _parameters[i-1]))
200  / (_parameters[i+1] - _parameters[i-1])
201  - sig * u[i-1].x()) / p);
202  // then y-value
203  p = sig * _derivatives[i-1].y() + 2.0;
204  p1.setY((sig - 1.0) / p);
205  p2.setY( (6.0 * ((_points[i+1].y() - _points[i].y())
206  / (_parameters[i+1] - _parameters[i])
207  - (_points[i].y() - _points[i-1].y())
208  / (_parameters[i] - _parameters[i-1]))
209  / (_parameters[i+1] - _parameters[i-1])
210  - sig * u[i-1].y()) / p);
211  // then z-value
212  p = sig * _derivatives[i-1].z() + 2.0;
213  p1.setZ((sig - 1.0) / p);
214  p2.setZ( (6.0 * ((_points[i+1].z() - _points[i].z())
215  / (_parameters[i+1] - _parameters[i])
216  - (_points[i].z() - _points[i-1].z())
217  / (_parameters[i] - _parameters[i-1]))
218  / (_parameters[i+1] - _parameters[i-1])
219  - sig * u[i-1].z()) / p);
220  _derivatives.push_back(p1);
221  u.push_back(p2);
222  }
223  }
224  }
225 
226  // BUGBUG: these are always 0, how does it work
227  qn = ZERO;
228  un = qn;
229 
230  QVector3D p3;
231  p3.setX(un.x() - qn.x() * u[_nopoints-2].x()
232  / (qn.x() * _derivatives[_nopoints-2].x() + 1.0));
233  p3.setY(un.y() - qn.y() * u[_nopoints-2].y()
234  / (qn.y() * _derivatives[_nopoints-2].y() + 1.0));
235  p3.setZ(un.z() - qn.z() * u[_nopoints-2].z()
236  / (qn.z() * _derivatives[_nopoints-2].z() + 1.0));
237  _derivatives.push_back(p3);
238 
239  // back substitution
240  for (int k=_nopoints-1; k>=1; --k) {
241  _derivatives[k-1] = _derivatives[k-1] * _derivatives[k] + u[k-1];
242  }
243  } // end _nopoints > 1
244  _build = true;
245  // determine min/max values
246  if (_nopoints > 0) {
247  for (size_t i=0; i<_nopoints; ++i) {
248  if (i == 0) {
249  _min = _points[i];
250  _max = _min;
251  }
252  else {
253  MinMax(_points[i], _min, _max);
254  }
255  }
256  }
257 }

259 QVector3D Spline::second_derive(float parameter)
260 {
261  QVector3D result = ZERO;
262  if (_nopoints < 2)
263  return result;
264  if (!_build)
265  rebuild();
266  if (_nopoints < 2)
267  return result;
268 
269  int lo, hi, k;
270 
271  if (_nopoints == 2) {
272  lo = 0;
273  hi = 1;
274  }
275  else {
276  lo = 0;
277  hi = _nopoints - 1;
278  do {
279  k = (lo + hi) / 2;
280  if (_parameters[k] < parameter)
281  lo = k;
282  else
283  hi = k;
284  } while (hi - lo > 1);
285  }
286  float frac;
287  if (_parameters[hi] - _parameters[lo] <= 0.0)
288  frac = 0.5;
289  else
290  frac = (parameter - _parameters[lo]) / (_parameters[hi] - _parameters[lo]);
291  result = _derivatives[lo] + (frac * (_derivatives[hi] - _derivatives[lo]));
292  return result;
293 }

295 float Spline::weight(size_t index)
296 {
297  float result;
298  float length, dist;
299 
300  if (index == 0 || (index == _nopoints - 1) || _knuckles[index]) {
301  result = 1E10;
302  }
303  else {
304  QVector3D p1 = _points[index-1];
305  QVector3D p2 = _points[index];
306  QVector3D p3 = _points[index+1];
307  length = (p3 - p1).length();
308  if (length < 1E-5) {
309  result = 0.0;
310  }
311  else {
312  dist = DistancepointToLine(p2,p1,p3);
313  if (dist < 1E-2) {
314  if (length*length/_total_length > 0.01)
315  result = 1E10;
316  else
317  result = 1E8 * dist * dist * length;
318  }
319  else
320  result = 1E8 * dist * dist * length;
321  }
322  }
323  return result;
324 }

348 bool Spline::simplify(float criterium)
349 {
350  vector<float> weights;
351  weights.reserve(_nopoints);
352  bool result = false;
353 
354  if (_nopoints < 3)
355  return true;
356 
357  float total_length = _total_length * _total_length;
358  if (total_length == 0)
359  return result;
360 
361  for (size_t i=0; i<_nopoints; ++i)
362  weights.push_back(weight(i)/total_length);
363 
364  vector<float>::iterator index;
365  do {
366  index = find_next_point(weights);
367  vector<float>::iterator last = weights.end();
368  last--;
369  if (index == weights.end() || _nopoints < 3 || index == last)
370  break;
371  if (*index < criterium) {
372  size_t j = index - weights.begin();
373  weights.erase(index);
374  _points.erase(_points.begin()+j);
375  _knuckles.erase(_knuckles.begin()+j);
376  _nopoints--;
377  if (j > 0 && j < _nopoints)
378  weights[j] = weights[j]/total_length;
379  }
380  else
381  break;
382  }
383  while (index != weights.end());
384  result = true;
385 
386  setBuild(false);
387 
388  return result;
389 }

419 float Spline::chord_length_approximation(float percentage)
420 {
421  float result, length, total_length;
422  float t1, t2;
423  t1 = 0;
424  t2 = 1.0;
425  float parameter = 1.0;
426  int counter = 0;
427  float l1 = 0;
428  float l2 = 0;
429  float desired_length = 0;
430  if (percentage < 0)
431  result = 0;
432  else if (percentage > 1)
433  result = 1;
434  else {
435  do {
436  length = coord_length(0, parameter);
437  if (counter == 0) {
438  l1 = 0;
439  l2 = length;
440  total_length = length;
441  desired_length = percentage*total_length;
442  parameter = percentage;
443  }
444  else {
445  if (length > desired_length) {
446  t2 = parameter;
447  l2 = length;
448  }
449  else {
450  t1 = parameter;
451  l1 = length;
452  }
453  parameter = t1 + ((desired_length - l1) / (l2-l1)) * (t2 - t1);
454  if (parameter < 0)
455  parameter = 0;
456  if (parameter > 1)
457  parameter = 1;
458  }
459  counter++;
460  }
461  while (counter <= 75 && fabs(length - desired_length) >= 1E-3);
462  result = parameter;
463  }
464  return result;
465 }

467 float Spline::curvature(float parameter, QVector3D& normal)
468 {
469  float result;
470 
471  QVector3D vel1 = first_derive(parameter);
472  QVector3D acc = second_derive(parameter);
473 
474  QVector3D crossproduct = QVector3D::crossProduct(acc, vel1);
475  float l = crossproduct.length();
476  if (l == 0)
477  l = 0.00001f;
478  float vdota = (vel1.x()*acc.x()) + (vel1.y()*acc.y()) + (vel1.z()*acc.z());
479  float vdotv = (vel1.x()*vel1.x()) + (vel1.y()*vel1.y()) + (vel1.z()*vel1.z());
480  if (vdotv < 0)
481  result = 0;
482  else
483  result = l/pow(vdotv, static_cast<float>(1.5));
484  normal = (vdotv * acc) - (vdota * vel1);
485  normal.normalize();
486  return result;
487 }

619 bool Spline::intersect_plane(const Plane& plane, IntersectionData& output)
620 {
621  output.number_of_intersections = 0;
622  float t1 = 0;
623  QVector3D p1 = value(t1);
624  float s1 = plane.distance(p1);
625  if (fabs(s1) < 1E-6)
626  add_to_output(p1, t1, output);
627  for (size_t i=1; i<=_fragments; ++i) {
628  float t2 = i / static_cast<float>(_fragments);
629  QVector3D p2 = value(t2);
630  float s2 = plane.distance(p2);
631  if (fabs(s2) < 1E-6)
632  add_to_output(p2, t2, output);
633  if ((s1 < 0 && s2 > 0) || (s2 < 0 && s1 > 0)) {
634  // intersection found
635  float t = s1 / (s2 - s1);
636  t = t1 + t * (t2 - t1);
637  add_to_output(value(t), t, output);
638  }
639  p1 = p2;
640  s1 = s2;
641  t1 = t2;
642  }
643  return output.number_of_intersections > 0;
644 }

754 QVector3D Spline::value(float parameter)
755 {
756  QVector3D result;
757  if (_nopoints < 2)
758  return result;
759  if (!_build)
760  rebuild();
761  if (_nopoints < 2)
762  return result;
763 
764  int lo, hi;
765  if (_nopoints == 2) {
766  lo = 0;
767  hi = 1;
768  }
769  else {
770  lo = 0;
771  hi = _nopoints - 1;
772  do {
773  int k = (lo + hi) / 2;
774  if (_parameters[k] < parameter)
775  lo = k;
776  else
777  hi = k;
778  }
779  while (hi - lo > 1);
780  }
781  float h, a, b;
782  h = _parameters[hi] - _parameters[lo];
783  if (fabs(h) < 1E-6) {
784  result = _points[hi];
785  }
786  else {
787  a = (_parameters[hi] - parameter) / h;
788  //b = (parameter - _parameters[lo]) / h;
789  b = 1 - a;
790  result = a * _points[lo] + b * _points[hi] + ((a * a * a - a) * _derivatives[lo]
791  + (b * b * b - b) * _derivatives[hi])
792  * (h * h) / 6.0;
793  }
794  return result;
795 }
