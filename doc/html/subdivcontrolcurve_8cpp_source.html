148 void SubdivisionControlCurve::deleteEdge(SubdivisionControlEdge* edge)
149 {
150  bool delcurve = false;
151  SubdivisionControlPoint* p1, *p2;
152  SubdivisionEdge* anedge;
153 
154  size_t i = 2;
155  while (i <= _points.size()) {
156  if ((_points[i-2] == edge->startPoint() && _points[i-1] == edge->endPoint())
157  || (_points[i-2] == edge->endPoint() && _points[i-1] == edge->startPoint())) {
158  // remove references to this curve from control edges
159  for (size_t j=2; j<=_points.size(); ++j) {
160  anedge = _owner->edgeExists(_points[j-2], _points[j-1]);
161  if (anedge != 0)
162  anedge->setCurve(0);
163  }
164  // remove references from subdivided edges
165  for (size_t j=2; j<=_div_points.size(); ++j) {
166  anedge = _owner->edgeExists(_div_points[j-2], _div_points[j-1]);
167  if (anedge != 0)
168  anedge->setCurve(0);
169  }
170  _div_points.clear();
171 
172  if (i - 2 > 0) {
173  // build first new curve
174  SubdivisionControlCurve* newcurve = SubdivisionControlCurve::construct(_owner);
175  _owner->addControlCurve(newcurve);
176  p1 = 0;
177  for (size_t j=0; j<=i-2; ++j) {
178  p2 = _points[j];
179  newcurve->addPoint(p2);
180  if (j > 0) {
181  anedge = _owner->edgeExists(p1, p2);
182  if (anedge != 0)
183  anedge->setCurve(newcurve);
184  }
185  p1 = p2;
186  }
187  newcurve->setSelected(isSelected());
188  }
189  if (i - 1 < _points.size() - 1) {
190  // build second new curve
191  SubdivisionControlCurve* newcurve = SubdivisionControlCurve::construct(_owner);
192  _owner->addControlCurve(newcurve);
193  p1 = 0;
194  for (size_t j=i-1; j<=_points.size()-1; ++j) {
195  p2 = _points[j];
196  newcurve->addPoint(p2);
197  if (j > i - 1) {
198  anedge = _owner->edgeExists(p1, p2);
199  if (anedge != 0)
200  anedge->setCurve(newcurve);
201  }
202  p1 = p2;
203  }
204  newcurve->setSelected(isSelected());
205  }
206  delcurve = true;
207  break;
208  }
209  ++i;
210  } // end while
211  if (delcurve) {
212  _points.clear();
213  if (_owner->hasSelectedControlCurve(this))
214  _owner->removeSelectedControlCurve(this);
215  if (_owner->hasControlCurve(this))
216  _owner->removeControlCurve(this);
217  // BUGBUG: we are supposed to delete this curve now, but can't
218  }
219 }

221 void SubdivisionControlCurve::draw(Viewport &vp, LineShader* lineshader)
222 {
223  if (numberOfControlPoints() <= 1)
224  return;
225  bool sel = isSelected();
226  Spline* curve = getSpline();
227  curve->setShowCurvature(sel && _owner->showCurvature());
228  if (curve->showCurvature())
229  curve->setFragments(600);
230  else
231  curve->setFragments(250);
232  curve->setCurvatureColor(_owner->getCurvatureColor());
233  curve->setCurvatureScale(_owner->getCurvatureScale());
234  QVector<QVector3D> vertices;
235  // BUGBUG: not implemented yet
236  if (!_owner->showControlNet() && sel) {
237  // draw controlpoints and edges
238  for (size_t i=2; i<=_points.size(); ++i) {
239  SubdivisionPoint* p1 = _points[i-2];
240  SubdivisionPoint* p2 = _points[i-1];
241  SubdivisionControlEdge* edge = _owner->controlEdgeExists(p1, p2);
242  if (edge != 0)
243  edge->draw(vp, lineshader);
244  if (i == 2)
245  vertices << p1->getCoordinate();
246  vertices << p2->getCoordinate();
247  }
248  lineshader->renderPoints(vertices, getColor());
249  }
250  vertices.clear();
251  if (vp.getViewportType() == Viewport::fvBodyplan && !_owner->drawMirror()) {
252  // draw mainframe location
253  Plane mfl(1, 0, 0, -_owner->getMainframeLocation());
254  IntersectionData output;
255  vector<float> parameters;
256  parameters.push_back(0);
257  parameters.push_back(1.0);
258  if (curve->intersect_plane(mfl, output)) {
259  parameters.insert(parameters.end(), output.parameters.begin(), output.parameters.end());
260  sort(parameters.begin(), parameters.end());
261  }
262  QVector<QVector3D> curvelines;
263  for (size_t i=2; i<=parameters.size(); ++i) {
264  QVector3D p3d = curve->value(0.5*(parameters[i-2] + parameters[i-1]));
265  int scale;
266  if (p3d.x() < _owner->getMainframeLocation())
267  scale = -1;
268  else
269  scale = 1;
270  size_t fragm = (parameters[i-1] - parameters[i-2]) * curve->getFragments();
271  if (fragm < 10)
272  fragm = 10;
273  if (curve->showCurvature()) {
274  for (size_t j=1; j<=fragm; ++j) {
275  float t = parameters[i-2] + (parameters[i-1] - parameters[i-2]) * (j - 1) / (fragm - 1);
276  QVector3D normal;
277  p3d = curve->value(t);
278  float c = curve->curvature(t, normal);
279  p3d.setY(p3d.y() * scale);
280  normal.setY(normal.y() * scale);
281  QVector3D p3d2 = p3d - (c * 2 * curve->getCurvatureScale() * normal);
282  vertices << p3d;
283  vertices << p3d2;
284  if ((j % 4) == 0 || j == 1 || j == fragm) {
285  // draw normal lines
286  curvelines << p3d;
287  curvelines << p3d2;
288  }
289  vertices << p3d2;
290  }
291  for (size_t j=0; j<curvelines.size(); ++j)
292  vertices << curvelines[j];
293  }
294  else {
295  vertices.clear();
296  for (size_t j=1; j<=fragm; ++j) {
297  float t = parameters[i-2] + (parameters[i-1] - parameters[i-2]) * (j - 1) / (fragm - 1);
298  p3d = curve->value(t);
299  p3d.setY(p3d.y() * scale);
300  vertices << p3d;
301  }
302  }
303  }
304  lineshader->renderLines(vertices, curve->getCurvatureColor());
305  }
306  else {
307  curve->draw(vp, lineshader);
308  }
309  if (_owner->drawMirror()) {
310  // draw reversed curve
311  for (size_t i=1; i<=curve->numberOfPoints(); ++i) {
312  QVector3D p3d = curve->getPoint(i-1);
313  p3d.setY(-p3d.y());
314  curve->setPoint(i-1, p3d);
315  }
316  curve->draw(vp, lineshader);
317  for (size_t i=1; i<=curve->numberOfPoints(); ++i) {
318  QVector3D p3d = curve->getPoint(i-1);
319  p3d.setY(-p3d.y());
320  curve->setPoint(i-1, p3d);
321  }
322  }
323 }
