150 void SubdivisionControlCurve::deleteEdge(SubdivisionControlEdge* edge)
151 {
152  bool delcurve = false;
153  SubdivisionControlPoint* p1, *p2;
154  SubdivisionEdge* anedge;
155 
156  size_t i = 2;
157  while (i <= _points.size()) {
158  if ((_points[i-2] == edge->startPoint() && _points[i-1] == edge->endPoint())
159  || (_points[i-2] == edge->endPoint() && _points[i-1] == edge->startPoint())) {
160  // remove references to this curve from control edges
161  for (size_t j=2; j<=_points.size(); ++j) {
162  anedge = _owner->edgeExists(_points[j-2], _points[j-1]);
163  if (anedge != 0)
164  anedge->setCurve(0);
165  }
166  // remove references from subdivided edges
167  for (size_t j=2; j<=_div_points.size(); ++j) {
168  anedge = _owner->edgeExists(_div_points[j-2], _div_points[j-1]);
169  if (anedge != 0)
170  anedge->setCurve(0);
171  }
172  _div_points.clear();
173 
174  if (i - 2 > 0) {
175  // build first new curve
176  SubdivisionControlCurve* newcurve = SubdivisionControlCurve::construct(_owner);
177  _owner->addControlCurve(newcurve);
178  p1 = 0;
179  for (size_t j=0; j<=i-2; ++j) {
180  p2 = _points[j];
181  newcurve->addPoint(p2);
182  if (j > 0) {
183  anedge = _owner->edgeExists(p1, p2);
184  if (anedge != 0)
185  anedge->setCurve(newcurve);
186  }
187  p1 = p2;
188  }
189  newcurve->setSelected(isSelected());
190  }
191  if (i - 1 < _points.size() - 1) {
192  // build second new curve
193  SubdivisionControlCurve* newcurve = SubdivisionControlCurve::construct(_owner);
194  _owner->addControlCurve(newcurve);
195  p1 = 0;
196  for (size_t j=i-1; j<=_points.size()-1; ++j) {
197  p2 = _points[j];
198  newcurve->addPoint(p2);
199  if (j > i - 1) {
200  anedge = _owner->edgeExists(p1, p2);
201  if (anedge != 0)
202  anedge->setCurve(newcurve);
203  }
204  p1 = p2;
205  }
206  newcurve->setSelected(isSelected());
207  }
208  delcurve = true;
209  break;
210  }
211  ++i;
212  } // end while
213  if (delcurve) {
214  _points.clear();
215  if (_owner->hasSelectedControlCurve(this))
216  _owner->removeSelectedControlCurve(this);
217  if (_owner->hasControlCurve(this))
218  _owner->removeControlCurve(this);
219  // BUGBUG: we are supposed to delete this curve now, but can't
220  }
221 }

223 void SubdivisionControlCurve::draw(Viewport &vp, LineShader* lineshader)
224 {
225  if (numberOfControlPoints() <= 1)
226  return;
227  bool sel = isSelected();
228  Spline* curve = getSpline();
229  curve->setColor(_owner->getControlCurveColor());
230  curve->setShowCurvature(sel && _owner->showCurvature());
231  if (curve->showCurvature())
232  curve->setFragments(600);
233  else
234  curve->setFragments(250);
235  curve->setCurvatureColor(_owner->getCurvatureColor());
236  curve->setCurvatureScale(_owner->getCurvatureScale());
237  QVector<QVector3D> vertices;
238  if (!_owner->showControlNet() && sel) {
239  // draw controlpoints and edges
240  for (size_t i=2; i<=_points.size(); ++i) {
241  SubdivisionPoint* p1 = _points[i-2];
242  SubdivisionPoint* p2 = _points[i-1];
243  SubdivisionControlEdge* edge = _owner->controlEdgeExists(p1, p2);
244  if (edge != 0)
245  edge->draw(vp, lineshader);
246  if (i == 2)
247  vertices << p1->getCoordinate();
248  vertices << p2->getCoordinate();
249  }
250  lineshader->renderPoints(vertices, getColor());
251  }
252  vertices.clear();
253  if (vp.getViewportType() == fvBodyplan && !_owner->drawMirror()) {
254 
255  vector<QVector3D> parray1;
256  vector<QVector3D> parray2;
257 
258  // draw mainframe location
259  Plane mfl(1, 0, 0, -_owner->getMainframeLocation());
260  IntersectionData output;
261  vector<float> parameters;
262  parameters.push_back(0);
263  parameters.push_back(1.0);
264  if (curve->intersect_plane(mfl, output)) {
265  parameters.insert(parameters.end(), output.parameters.begin(), output.parameters.end());
266  sort(parameters.begin(), parameters.end());
267  }
268  QVector<QVector3D> curvelines;
269  for (size_t i=2; i<=parameters.size(); ++i) {
270  QVector3D p3d = curve->value(0.5*(parameters[i-2] + parameters[i-1]));
271  int scale;
272  if (p3d.x() < _owner->getMainframeLocation())
273  scale = -1;
274  else
275  scale = 1;
276  size_t fragm = (parameters[i-1] - parameters[i-2]) * curve->getFragments();
277  if (fragm < 10)
278  fragm = 10;
279  for (size_t j=0; j<fragm; j++) {
280  float t = parameters[i-1] + (parameters[i] - parameters[i-1]) * j / (fragm - 1);
281  QVector3D p = curve->value(t);
282  p.setY(p.y() * scale);
283  parray1.push_back(p);
284  }
285  if (curve->showCurvature()) {
286  for (size_t j=0; j<fragm; ++j) {
287  float t = parameters[i-2] + (parameters[i-1] - parameters[i-2]) * j / (fragm - 1);
288  QVector3D normal;
289  float c = curve->curvature(t, normal);
290  normal.setY(normal.y() * scale);
291  QVector3D p3d2 = parray1[j] - (c * 2 * curve->getCurvatureScale() * normal);
292  parray2.push_back(p3d2);
293  }
294  for (size_t j=1; j<=fragm; j++) {
295  if ((j % 4) == 0 || j == 1 || j == fragm) {
296  // draw normal lines
297  vertices << parray1[j-1];
298  vertices << parray2[j-1];
299  }
300  }
301  for (size_t j=1; j<parray2.size(); j++) {
302  vertices << parray2[j-1];
303  vertices << parray2[j];
304  }
305  glLineWidth(1);
306  lineshader->renderLines(vertices, curve->getCurvatureColor());
307  }
308  else {
309  for (size_t j=1; j<=parray1.size(); ++j) {
310  vertices << parray1[j-1];
311  vertices << parray1[j];
312  }
313  glLineWidth(1);
314  lineshader->renderLines(vertices, getColor());
315  }
316  }
317  }
318  else {
319  curve->draw(vp, lineshader);
320  }
321  if (_owner->drawMirror()) {
322  // draw reversed curve
323  for (size_t i=1; i<=curve->numberOfPoints(); ++i) {
324  QVector3D p3d = curve->getPoint(i-1);
325  p3d.setY(-p3d.y());
326  curve->setPoint(i-1, p3d);
327  }
328  curve->draw(vp, lineshader);
329  for (size_t i=1; i<=curve->numberOfPoints(); ++i) {
330  QVector3D p3d = curve->getPoint(i-1);
331  p3d.setY(-p3d.y());
332  curve->setPoint(i-1, p3d);
333  }
334  }
335 }

361 void SubdivisionControlCurve::loadBinary(FileBuffer &source)
362 {
363  quint32 n, ind;
364  source.load(n);
365  SubdivisionControlPoint* p1 = 0;
366  for (size_t i=1; i<=n; ++i) {
367  source.load(ind);
