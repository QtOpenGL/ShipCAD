167  intersection_type_t _intersection_type;
168  Plane _plane;
169  CalculateSplineArea(float* area, QVector3D* cog, QVector2D* mom_inertia, intersection_type_t ty, const Plane& pln)
170  : _area(area), _cog(cog), _moi(mom_inertia), _intersection_type(ty), _plane(pln)

172  QVector2D ProjectTo2D(QVector3D& p)
173  {
174  QVector2D result;
175  switch (_intersection_type) {
176  case fiStation:
177  result = QVector2D(p.y(), p.z());
178  break;
179  case fiButtock:
180  result = QVector2D(p.x(), p.z());
181  break;
182  case fiWaterline:
183  result = QVector2D(p.x(), p.y());
184  break;
185  default:
186  result = ZERO2;
187  }
188  return result;
189  }

191  void operator()(Spline* spline)
192  {
193  bool closed_spline;
194  IntersectionData intersection_data;
195  vector<float> parameters;
196  *_area = 0;
197  *_cog = ZERO;
198  *_moi = ZERO2;
199  QVector2D c = ZERO2;
200  QVector2D momi = c;
201  closed_spline = DistPP3D(spline->getPoint(0), spline->getPoint(spline->numberOfPoints()-1)) < 1e-4;
202  if (!closed_spline) {
203  // make it closed
204  spline->add(spline->getPoint(0));
205  spline->setKnuckle(spline->numberOfPoints()-2, true);
206  }
207  spline->setFragments(500);
208  parameters.push_back(0);
209  parameters.push_back(1.0);
210  if (_intersection_type != fiWaterline) {
211  if (spline->intersect_plane(_plane, intersection_data)) {
212  for (size_t i=0; i<intersection_data.number_of_intersections; i++)
213  parameters.push_back(intersection_data.parameters[i]);
214  }
215  }
216  sort(parameters.begin(), parameters.end());
217  if (parameters.size()) {
218  float t1 = 0;
219  for (size_t i=2; i<parameters.size(); i++) {
220  float t2 = parameters[i-1];
221  float t = 0.5 * (t1 + t2);
222  QVector3D p = spline->value(t);
223  // check on which side of the plane this point is
224  float side = _plane.distance(p);
225  if (side < 0 || _intersection_type == fiWaterline) {
226  // the point lies at the back of the plane, include this area
227  QVector2D p1 = ZERO2;
228  for (size_t j=0; j<500; j++) {
229  t = t1 + (j/500.0) * (t2-t1);
230  p = spline->value(t);
231  QVector2D p2 = ProjectTo2D(p);
232  if (j > 0) {
233  float delta = 0.5 * (p2.x() + p1.x()) * (p2.y() - p1.y());
234  *_area += delta;
235  c.setX(c.x() + delta * .25 * (p2.x() + p1.x()));
236  c.setY(c.y() + delta * .5 * (p2.y() + p1.y()));
237  momi.setX(momi.x() + (1/12.) * (p1.y()+p2.y())*(p1.y()*p1.y()+p2.y()*p2.y())*(p2.x()-p1.x()));
238  momi.setY(momi.y() + (1/12.) * (p2.x()*p2.x()*(3*p2.y()+p1.y())+2*p1.x()*p2.x()*(p1.y()+p2.y())+p1.x()*p1.x()*(3*p1.y()+p2.y()))*(p2.x()-p1.x()));
239  }
240  p1 = p2;
241  }
242  }
243  t1 = t2;
244  }
245  if (_area != 0) {
246  c /= *_area;
247  momi.setX(fabs(momi.x()));
248  momi.setY(fabs(momi.y()));
249  switch (_intersection_type) {
250  case fiStation:
251  _cog->setX(-_plane.d());
252  _cog->setY(c.x());
253  _cog->setZ(c.y());
254  _moi->setX(0);
255  _moi->setY(momi.x());
256  break;
257  case fiButtock:
258  _cog->setX(c.x());
259  _cog->setY(-_plane.d());
260  _cog->setZ(c.y());
261  _moi->setX(momi.x());
262  _moi->setY(0);
263  break;
264  case fiWaterline:
265  _cog->setX(c.x());
266  _cog->setY(c.y());
267  _cog->setZ(-_plane.d());
268  _moi->setX(momi.x());
269  _moi->setY(momi.y());
270  break;
271  }
272  }
273  }
274  }

277 void Intersection::calculateArea(const Plane& plane, float* area, QVector3D* cog, QVector2D* moment_of_inertia)
278 {
279  *area = 0;
280  *cog = ZERO;
281  *moment_of_inertia = ZERO2;
282  QVector2D momi;
283  if (!isBuild())
284  rebuild();
285  if (_items.size() > 0) {
286  createStarboardPart(); // this ensures correct winding order
287  CalculateSplineArea calc(area, cog, moment_of_inertia, _intersection_type, plane);
288  for_each(_items.begin(), _items.end(), calc);
289  if (*area != 0) {
290  *cog = *cog / *area;
291  if (_intersection_type == fiWaterline) {
292  moment_of_inertia->setX(moment_of_inertia->x() - cog->y() * cog->y() * *area);
293  moment_of_inertia->setY(moment_of_inertia->y() - cog->x() * cog->x() * *area);
294  }
295  }
296  }
297  if (_items.size() == 0 || *area == 0) {
298  *cog = ZERO;
299  switch (_intersection_type) {
300  case fiStation:
301  cog->setX(-_plane.d());
302  break;
303  case fiButtock:
304  cog->setY(-_plane.d());
305  break;
306  case fiWaterline:
307  cog->setZ(-_plane.d());
308  break;
309  }
310  }

313 void Intersection::createStarboardPart()
314 {
315  if (_intersection_type != fiButtock) {
316  // copy all present splines and mirror the y coordinate
317  for (size_t i=_items.size(); i>=1; i--) {
318  Spline* spline = new Spline(*_items.get(i-1));
319  for (size_t j=0; j<spline->numberOfPoints(); j++) {
320  QVector3D p1 = spline->getPoint(j);
321  p1.setY(-p1.y());
322  spline->setPoint(j, p1);
323  }
324  _items.add(spline);
325  }
326  // try to connect the splines
327  JoinSplineSegments(0.05f, false, _items);
328  }
329  // check if the orientation is counterclockwise
330  for (size_t i=_items.size(); i>=1; i--) {
331  Spline* spline = _items.get(i-1);
332  float area = 0;
333  QVector3D p1 = spline->getPoint(spline->numberOfPoints()-1);
334  for (size_t j=0; j<500; j++) {
335  QVector3D p2 = spline->value(j/500.);
336  float delta;
337  switch (_intersection_type) {
338  case fiStation:
339  delta = 0.5 * (p2.y() + p1.y()) * (p2.z() - p1.z());
340  break;
341  case fiButtock:
342  delta = 0.5 * (p2.x() + p1.x()) * (p2.z() - p1.z());
343  break;
344  case fiWaterline:
345  delta = 0.5 * (p2.x() + p1.x()) * (p2.y() - p1.y());
346  break;
347  default:
348  throw runtime_error("66");
349  }
350  area += delta;
351  p1 = p2;
352  }
353  if (fabs(area) < 1e-4 && _items.size() > 0) {
354  // either this spline has a very small area (.01m x .01m) or it is an
355  // unconnected straight line. In both cases it may be deleted as long
356  // as at least one valid spline segment remains
357  _items.del(spline);
358  } else if (area < 0) {
359  // spline is defined clockwise, so invert the controlpoints
360  spline->invert_direction();
361  }
362  }

370 void Intersection::loadBinary(FileBuffer& source)
371 {
372  quint32 n;
373  QVector3D p;
374  source.load(n);
375  _intersection_type = static_cast<intersection_type_t>(n);
376  if (_owner->getFileVersion() >= fv191)
377  source.load(_show_curvature);
378  else
379  _show_curvature = false;
380  source.load(_plane);
381  source.load(_build);
382  source.load(n);
383  for (size_t i=0; i<n; i++) {
384  Spline* sp = new Spline();
385  _items.add(sp);
386  // read number of points for this spline
387  quint32 m;
388  source.load(m);
389  for (size_t j=0; j<m; j++) {
390  float x, y, z;
391  if (_owner->getFileVersion() >= fv160) {
392  if (_intersection_type == fiStation) {
393  p.setX(-_plane.d());
394  source.load(y);
395  p.setY(y);
396  source.load(z);
397  p.setZ(z);
398  } else if (_intersection_type == fiButtock) {
399  source.load(x);
400  p.setX(x);
401  p.setY(-_plane.d());
402  source.load(z);
403  p.setZ(z);
404  } else if (_intersection_type == fiWaterline) {
405  source.load(x);
406  p.setX(x);
407  source.load(y);
408  p.setY(y);
409  p.setZ(-_plane.d());
410  } else {
411  source.load(p);
412  }
413  } else {
414  source.load(p);
415  }
416  sp->add(p);
417  bool b;
418  source.load(b);
419  sp->setKnuckle(j, b);
420  }
421  }
422 }
