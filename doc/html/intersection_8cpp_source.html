115 void Intersection::rebuild()
116 {
117  _build = false;
118  _owner->getSurface()->intersectPlane(_plane, _use_hydrostatic_surfaces_only, _items);
119  // use a low simplification factor to remove only points that are (nearly) on a line
120  if (_owner->getProjectSettings().getSimplifyIntersections()) {
121  for (size_t i=0; i<_items.size(); i++)
122  _items[i]->simplify(2.0);
123  }
124  _build = true;
125 }

151 void Intersection::loadBinary(FileBuffer& source)
152 {
153  size_t i;
154  QVector3D p;
155  source.load(i);
156  _intersection_type = static_cast<intersection_type_t>(i);
157  if (_owner->getFileVersion() >= fv191)
158  source.load(_show_curvature);
159  else
160  _show_curvature = false;
161  source.load(_plane);
162  source.load(_build);
163  size_t n;
164  source.load(n);
165  for (i=0; i<n; i++) {
166  Spline* sp = new Spline();
167  _items.push_back(sp);
168  // read number of points for this spline
169  size_t m;
170  source.load(m);
171  for (size_t j=0; j<m; j++) {
172  float x, y, z;
173  if (_owner->getFileVersion() >= fv160) {
174  if (_intersection_type == fiStation) {
175  p.setX(-_plane.d());
176  source.load(y);
177  p.setY(y);
178  source.load(z);
179  p.setZ(z);
180  } else if (_intersection_type == fiButtock) {
181  source.load(x);
182  p.setX(x);
183  p.setY(-_plane.d());
184  source.load(z);
185  p.setZ(z);
186  } else if (_intersection_type == fiWaterline) {
187  source.load(x);
188  p.setX(x);
189  source.load(y);
190  p.setY(y);
191  p.setZ(-_plane.d());
192  } else {
193  source.load(p);
194  }
195  } else {
196  source.load(p);
197  }
198  sp->setPoint(j, p);
199  bool b;
200  source.load(b);
201  sp->setKnuckle(j, b);
202  }
203  }
204 }

206 void Intersection::saveBinary(FileBuffer& dest)
207 {
208  dest.add(static_cast<int>(_intersection_type));
209  if (_owner->getFileVersion() >= fv191)
210  dest.add(_show_curvature);
211  dest.add(_plane);
212  dest.add(_build);
213  dest.add(_items.size());
214  for (size_t i=0; i<_items.size(); i++) {
215  Spline* sp = _items[i];
216  dest.add(sp->numberOfPoints());
217  for (size_t j=0; j<sp->numberOfPoints(); j++) {
218  QVector3D p = sp->getPoint(j);
219  if (_owner->getFileVersion() >= fv160) {
220  switch (_intersection_type) {
221  case fiStation:
222  dest.add(p.y());
223  dest.add(p.z());
224  dest.add(sp->isKnuckle(j));
225  break;
226  case fiButtock:
227  dest.add(p.x());
228  dest.add(p.z());
229  dest.add(sp->isKnuckle(j));
230  break;
231  case fiWaterline:
232  dest.add(p.x());
233  dest.add(p.y());
234  dest.add(sp->isKnuckle(j));
235  break;
236  case fiDiagonal:
237  dest.add(p);
238  dest.add(sp->isKnuckle(j));
239  break;
240  }
241  } else {
242  dest.add(p);
243  dest.add(sp->isKnuckle(j));
244  }
245  }
246  }
