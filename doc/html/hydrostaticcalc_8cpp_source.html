196 // used in balance
197 static float DisplInterpolate(float x, float x1, float y1, float x2, float y2)
198 {
199  float result;
200 
201  if (fabs(x2 - x1) < 1e-3)
202  result = 0.5 * (y1 + y2);
203  else
204  result = y1 + (y2 - y1) * (x - x1) / (x2 - x1);
205  if (result < y1 || result > y2)
206  result = 0.5 * (y1 + y2);
207  return result;
208 }

239 static void CalculateMinMaxData(MinMaxData& mmd, ShipCADModel* owner,
240  Plane& wlplane, float CosTrim, float SinTrim,
241  float CosHeel, float SinHeel)
242 {
243  // zero out min max data
244  mmd.lowest_point = ZERO;
245  mmd.lowest_leak = ZERO;
246  mmd.plane_normal = ZERO;
247  mmd.max_draft = mmd.lowest_z = 0.0;
248  mmd.calculated = false;
249 
250  QVector3D p1 = RotatePointBack(ZERO, CosTrim, SinTrim, CosHeel, SinHeel);
251  QVector3D p2 = RotatePointBack(QVector3D(1.0, 0.0, 0.0),
252  CosTrim, SinTrim, CosHeel, SinHeel);
253  QVector3D p3 = RotatePointBack(QVector3D(1.0, 1.0, 0.0),
254  CosTrim, SinTrim, CosHeel, SinHeel);
255  // the following plane has the right orientation for the given trim
256  // and angle of heel, however the correct distance from the lowest point
257  // on the hull has to be calculated
258  mmd.plane_normal = UnifiedNormal(p1, p2, p3);
259  wlplane = Plane(p1, p2, p3);
260  bool first = true;
261  bool firstleak = true;
262  float min = 0;
263  float max = 0;
264  float distance;
265  for (size_t i=0; i<owner->numberOfLayers(); i++) {
266  if (owner->getLayer(i)->useInHydrostatics()) {
267  SubdivisionLayer* layer = owner->getLayer(i);
268  for (size_t j=0; j<layer->numberOfFaces(); j++) {
269  SubdivisionControlFace* face = layer->getFace(j);
270  for (size_t k=0; k<face->numberOfChildren(); k++) {
271  SubdivisionFace* child = face->getChild(k);
272  for (size_t l=0; l<child->numberOfPoints(); l++) {
273  QVector3D p = child->getPoint(l)->getCoordinate();
274  distance = wlplane.a() * p1.x() + wlplane.b() * p1.y()
275  + wlplane.c() * p1.z() + wlplane.d();
276  if (first) {
277  min = distance;
278  max = min;
279  mmd.lowest_point = p1;
280  mmd.lowest_z = p1.z();
281  } else {
282  if (distance < min) {
283  min = distance;
284  mmd.lowest_point = p1;
285  } else if (distance > max) {
286  max = distance;
287  }
288  if (p1.z() < mmd.lowest_z)
289  mmd.lowest_z = p1.z();
290  }
291  p1.setY(-p1.y());
292  distance = wlplane.a() * p1.x() + wlplane.b() * p1.y()
293  + wlplane.c() * p1.z() + wlplane.d();
294  if (distance < min) {
295  min = distance;
296  mmd.lowest_point = p1;
297  } else if (distance > max)
298  max = distance;
299  // check if this point is a leak point
300  if (fabs(p.y()) > 1e-4 && child->getPoint(l)->isBoundaryVertex()) {
301  p1 = p;
302  if (firstleak) {
303  firstleak = false;
304  mmd.lowest_leak = p1;
305  } else {
306  distance = wlplane.a() * p1.x()
307  + wlplane.b() * p1.y()
308  + wlplane.c() * p1.z() + wlplane.d();
309  float tmp = wlplane.a() * mmd.lowest_leak.x()
310  + wlplane.b() * mmd.lowest_leak.y()
311  + wlplane.c() * mmd.lowest_leak.z()
312  + wlplane.d();
313  if (distance < tmp)
314  mmd.lowest_leak = p1;
315  }
316  p1.setY(-p1.y());
317  distance = wlplane.a() * p1.x()
318  + wlplane.b() * p1.y()
319  + wlplane.c() * p1.z() + wlplane.d();
320  float tmp = wlplane.a() * mmd.lowest_leak.x()
321  + wlplane.b() * mmd.lowest_leak.y()
322  + wlplane.c() * mmd.lowest_leak.z()
323  + wlplane.d();
324  if (distance < tmp)
325  mmd.lowest_leak = p1;
326  }
327  }
328  }
329  }
330  }
331  }
332  if (!firstleak) {
333  // leak points have been found, check if this restricts the max draft
334  distance = wlplane.a() * mmd.lowest_leak.x()
335  + wlplane.b() * mmd.lowest_leak.y()
336  + wlplane.c() * mmd.lowest_leak.z()
337  + wlplane.d();
338  if (distance < max)
339  max = distance;
340  mmd.max_draft = max - min - 1e-4;
341  }
342 }

350 bool HydrostaticCalc::balance(float displacement, bool freetotrim,
351  CrosscurvesData& output)
352 {
353  if (displacement == 0)
354  return true;
355 
356  bool result = false;
357  int max_iterations = 25;
358  float max_error = 5e-4f;
359  float max_trim_error = 1e-4f;
360  int trim_iteration = 0;
361  int displ_iteration;
362  float cos_heel;
363  float sin_heel;
364  float cos_trim;
365  float sin_trim;
366  float error = 0;
367  float trim_error = 0;
368  float error_difference;
369  float prev_error;
370  Plane wlplane;
371  MinMaxData mmd;
372  DraftData min_draft;
373  DraftData max_draft;
374  DraftData curr_draft;
375  _data.clear();
376  mmd.calculated = false;
377  output.clear();
378 
379  do {
380  trim_iteration++;
381  cos_heel = cos(DegToRad(-_heeling_angle));
382  sin_heel = sin(DegToRad(-_heeling_angle));
383  cos_trim = cos(DegToRad(getTrimAngle()));
384  sin_trim = sin(DegToRad(getTrimAngle()));
385  if (!mmd.calculated)
386  CalculateMinMaxData(mmd, _owner, wlplane, cos_trim, sin_trim,
387  cos_heel, sin_heel);
388  min_draft.draft = 0;
389  min_draft.displ = 0;
390  max_draft.draft = mmd.max_draft;
391  wlplane = CalculateWaterlinePlane(max_draft.draft, mmd);
392  calculateVolume(wlplane);
393  max_draft.displ = _data.displacement;
394  if (displacement > 1.005 * max_draft.displ)
395  addError(feNotEnoughBuoyancy);
396  else {
397  displ_iteration = 0;
398  prev_error = 0;
399  do {
400  displ_iteration++;
401  curr_draft.draft = DisplInterpolate(displacement,
402  min_draft.displ,
403  min_draft.draft,
404  max_draft.displ,
405  max_draft.draft);
406  wlplane = CalculateWaterlinePlane(curr_draft.draft, mmd);
407  calculateVolume(wlplane);
408  curr_draft.displ = _data.displacement;
409  if (displacement < 0.1)
410  error = fabs(displacement - curr_draft.displ);
411  else
412  error = fabs((displacement - curr_draft.displ) / displacement);
413  if (error > max_error) {
414  if (curr_draft.displ < displacement)
415  min_draft = curr_draft;
416  else
417  max_draft = curr_draft;
418  }
419  error_difference = fabs(error - prev_error);
420  prev_error = error;
421  } while ((error >= max_error)
422  && (displ_iteration <= max_iterations)
423  && (error_difference >= 1e-5));
424  }
425  if (freetotrim) {
426  // TODO
427  } else
428  trim_error = 0;
429  } while ((trim_iteration <= max_iterations)
430  && (trim_error <= max_trim_error));
431  result = (trim_iteration > max_iterations) && (error <= max_error)
432  && (error_difference < 1e-5);
433  if (result) {
434  output.waterline_plane = _data.waterline_plane;
435  output.absolute_draft = _data.absolute_draft;
436  output.volume = _data.volume;
437  output.displacement = _data.displacement;
438  output.center_of_buoyancy = _data.center_of_buoyancy;
439  if (fabs(_heeling_angle) < 1e-5)
440  output.center_of_buoyancy.setY(0.0);
441  }
442  return result;
443 }

506  void ProcessTriangle(QVector3D& p1, QVector3D& p2, QVector3D& p3)
507  {
508  QVector3D volume_moment;
509  // reposition points with respect to the new projected origin
510  p1 -= new_origin;
511  p2 -= new_origin;
512  p3 -= new_origin;
513  QVector3D center((p1.x() + p2.x() + p3.x()) / 3,
514  (p1.y() + p2.y() + p3.y()) / 3,
515  (p1.z() + p2.z() + p3.z()) / 3);
516  float volume = ((p1.z() * (p2.x() * p3.y() - p2.y() * p3.x()))
517  + (p1.y() * (p2.z() * p3.x() - p2.x() * p3.z()))
518  + (p1.x() * (p2.y() * p3.z() - p2.z() * p3.y()))) / 6;
519  if (volume != 0) {
520  volume_moment = .75 * center * volume;
521  data.volume += volume;
522  data.center_of_buoyancy += volume_moment;
523  }
524  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z()) + (p2.y() - p3.y()) * (p2.z() + p3.z())
525  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
526  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x()) + (p2.z() - p3.z()) * (p2.x() + p3.x())
527  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
528  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y()) + (p2.x() - p3.x()) * (p2.y() + p3.y())
529  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
530  data.wetted_surface += sqrt(ax * ax + ay * ay + az * az);
531  }

533  void run() {
534  QVector3D p, p1, p2, p3;
535  bool submerged = false;
536  float side1, side2, parameter;
537  vector<QVector3D> points;
538  for (size_t i=0; i<hydro_calc->getOwner()->numberOfLayers(); i++) {
539  SubdivisionLayer* layer = hydro_calc->getOwner()->getLayer(i);
540  if (!layer->useInHydrostatics()) continue;
541  for (size_t j=0; j<layer->numberOfFaces(); j++) {
542  SubdivisionControlFace* face = layer->getFace(j);
543  for (size_t k=0; k<face->numberOfChildren(); k++) {
544  SubdivisionFace* child = face->getChild(k);
545  // calculate the portside of the model
546  points.clear();
547  p1 = child->getPoint(child->numberOfPoints()-1)->getCoordinate();
548  // calculate on which side of the waterplane this point is
549  side1 = data.waterline_plane.distance(p1);
550  for (size_t l=0; l<child->numberOfPoints(); l++) {
551  p2 = child->getPoint(l)->getCoordinate();
552  side2 = data.waterline_plane.distance(p2);
553  if ((side1 < -1e-5 && side2 > 1e-5) || (side1 > 1e-5 && side2 < -1e-5)) {
554  // the current linesegment between p1-p2 intersects the waterline plane
555  if (side1 == side2)
556  parameter = .5 * (side1 + side2);
557  else
558  parameter = -side1 / (side2 - side1);
559  p = p1 + parameter * (p2 - p1);
560  points.push_back(p);
561  CheckSubmergedBody(p, 0);
562  }
563  if (side2 <= 1e-5) {
564  if (side2 < data.absolute_draft)
565  data.absolute_draft = side2;
566  // p2 lies also on or under the waterlineplane
567  points.push_back(p2);
568  if (side2 < -1e-5) {
569  // point is submerged, check if the model is making water
570  if (child->getPoint(l)->isBoundaryVertex() && fabs(child->getPoint(l)->getCoordinate().y()) > 1e-4) {
571  if (!hydro_calc->hasError(feMakingWater)) {
572  hydro_calc->addError(feMakingWater);
573  data.leak = child->getPoint(l)->getCoordinate();
574  }
575  }
576  }
577  CheckSubmergedBody(p2, side2);
578  }
579  p1 = p2;
580  side1 = side2;
581  } // end of child face point loop
582 
583  // calculate volume aft of this face
584  if (points.size() > 2)
585  submerged = true;
586  for (size_t l=3; l<points.size(); l++)
587  ProcessTriangle(points[0], points[l-2], points[l-1]);
588  if (layer->isSymmetric()) {
589  // calculate the starboard side of the model
590  points.clear();
591  p1 = child->getPoint(child->numberOfPoints()-1)->getCoordinate();
592  p1.setY(-p1.y());
593  // calculate on which side of the waterplane this point is
594  side1 = data.waterline_plane.distance(p1);
595  for (size_t l=0; l<child->numberOfPoints(); l++) {
596  p2 = child->getPoint(l)->getCoordinate();
597  side2 = data.waterline_plane.distance(p2);
598  if ((side1 < -1e-5 && side2 > 1e-5) || (side1 > 1e-5 && side2 < -1e-5)) {
599  // the current linesegment between p1-p2 intersects the waterline plane
600  if (side1 == side2)
601  parameter = .5 * (side1 + side2);
602  else
603  parameter = -side1 / (side2 - side1);
604  p = p1 + parameter * (p2 - p1);
605  points.push_back(p);
606  CheckSubmergedBody(p, 0);
607  }
608  if (side2 <= 1e-5) {
609  if (side2 < data.absolute_draft)
610  data.absolute_draft = side2;
611  // p2 lies also on or under the waterlineplane
612  points.push_back(p2);
613  if (side2 < -1e-5) {
614  // point is submerged, check if the model is making water
615  if (child->getPoint(l)->isBoundaryVertex() && fabs(child->getPoint(l)->getCoordinate().y()) > 1e-4) {
616  if (!hydro_calc->hasError(feMakingWater)) {
617  hydro_calc->addError(feMakingWater);
618  data.leak = child->getPoint(l)->getCoordinate();
619  }
620  }
621  }
622  CheckSubmergedBody(p2, side2);
623  }
624  p1 = p2;
625  side1 = side2;
626  } // end of face point loop
627  // calculate volume aft of this face
628  for (size_t l=3; l<points.size(); l++)
629  ProcessTriangle(points[0], points[l-2], points[l-1]);
630  } // end of symmetric face if
631  } // end of face loop
632  } // end of subdiv control face loop
633  } // end of layers
634 
635  data.absolute_draft = -data.absolute_draft;
636  if (first_point) {
637  // no intersection with the watersurface found, the ship is either
638  // not submerged or totally submerged
639  if (!submerged) {
640  hydro_calc->addError(feNothingSubmerged);
641  data.absolute_draft = 0;
642  }
643  }
644  if (hydro_calc->hasError(feMakingWater)) {
645  data.volume = 0;
646  data.center_of_buoyancy = ZERO;
647  }
648 
649  ProjectSettings& ps = hydro_calc->getOwner()->getProjectSettings();
650 
651  data.displacement = VolumeToDisplacement(data.volume,
652  ps.getWaterDensity(),
653  ps.getAppendageCoefficient(),
654  ps.getUnits());
655  data.length_waterline = data.wl_max.x() - data.wl_min.x();
656  data.beam_waterline = data.wl_max.y() - data.wl_min.y();
657  if (data.volume != 0) {
658  // translate center of buoyancy back to the original origin
659  data.center_of_buoyancy = new_origin + data.center_of_buoyancy / data.volume;
660  data.center_of_buoyancy = RotatePoint(data.center_of_buoyancy);
661  if (data.length_waterline != 0) {
662  data.lcb_perc = 100 * data.center_of_buoyancy.x() - ps.getMainframeLocation() / data.length_waterline;
663  }
664  // apply appendage coeff
665  data.volume *= ps.getAppendageCoefficient();
666  }
667  }
668 };

694 void HydrostaticCalc::calculate()
695 {
696  QVector3D p, p1, p2, p3;
697  bool first_point = true;
698  float parameter;
699 
700  setCalculated(false);
701 
702  // calculate overall extents of the hull alone
703  for (size_t i=0; i<_owner->numberOfLayers(); i++) {
704  SubdivisionLayer* layer = _owner->getLayer(i);
705  if (layer->useInHydrostatics()) {
706  for (size_t j=0; j<layer->numberOfFaces(); j++) {
707  SubdivisionControlFace* face = layer->getFace(j);
708  for (size_t k=0; k<face->numberOfChildren(); k++) {
709  SubdivisionFace* child = face->getChild(k);
710  for (size_t l=0; l<child->numberOfPoints(); l++) {
711  p2 = child->getPoint(l)->getCoordinate();
712  if (first_point) {
713  _data.model_min = p2;
714  _data.model_max = p2;
715  first_point = false;
716  } else {
717  MinMax(p2, _data.model_min, _data.model_max);
718  }
719  }
720  }
721  }
722  }
723  }
724 
725  // setup volume calculation using our waterline plane
726  VolumeCalc vc(getWlPlane(), this);
727  vc.run();
728 
729  ProjectSettings& ps = _owner->getProjectSettings();
730 
731  float submerged_length = _data.sub_max.x() - _data.sub_min.x();
732  float submerged_width = _data.sub_max.y() - _data.sub_min.y();
733 
734  QVector2D tmpp2d;
735 
736  // calculate mainframe properties
737  if (_data.volume > 0 && _errors.size() == 0 && (hasCalculation(hcMainframe) || hasCalculation(hcAll))) {
738  _mainframe->setIntersectionType(fiStation);
739  _mainframe->setUseHydrostaticsSurfacesOnly(true);
740  _mainframe->setPlane(Plane(1,0,0,ps.getMainframeLocation()));
741  _mainframe->calculateArea(_data.waterline_plane, &_data.mainframe_area, &_data.mainframe_cog, &tmpp2d);
742  _data.mainframe_cog.setZ(_data.mainframe_cog.z() - _data.model_min.z());
743  if (ps.getHydrostaticCoefficients() == fcActualData) {
744  if (submerged_width * _draft != 0.0) {
745  _data.mainframe_coeff = _data.mainframe_area / (submerged_width * _draft);
746  }
747  } else if (ps.getBeam() * _draft != 0) {
748  _data.mainframe_coeff = _data.mainframe_area / (ps.getBeam() * _draft);
749  }
750  }
751 
752  // calculate waterline properties
753  if (_data.volume > 0 && _errors.size() == 0 && (hasCalculation(hcWaterline) or hasCalculation(hcAll))) {
754  Intersection* waterplane = new Intersection(_owner, fiWaterline, _data.waterline_plane, true);
755  waterplane->rebuild();
756  parameter = -1e6;
757  _data.waterplane_entrance_angle = 0;
758  SplineVector& wpsplines = waterplane->getSplines();
759  for (size_t j=0; j<wpsplines.size(); j++) {
760  Spline* spline = wpsplines.get(j);
761  // rotate all points back to a horizontal plane
762  for (size_t k=0; k<spline->numberOfPoints(); k++) {
763  p1 = spline->getPoint(k);
764  p2 = vc.RotatePoint(p1);
765  spline->setPoint(k, p2);
766  }
767  if (spline->value(0).x() > spline->value(1).x())
768  spline->invert_direction();
769  p1 = spline->value(1);
770  if (p1.x() > parameter) {
771  parameter = p1.x();
772  p2 = spline->value(0.99);
773  if (p1.x() - p2.x() != 0)
774  _data.waterplane_entrance_angle = RadToDeg(atan((p2.y() - p1.y()) / (p1.x() - p2.x())));
775  else
776  _data.waterplane_entrance_angle = -_data.waterplane_entrance_angle;
777  }
778  }
779  waterplane->calculateArea(_data.waterline_plane, &_data.waterplane_area, &_data.waterplane_cog, &_data.waterplane_mom_inertia);
780  if (ps.getBeam() * ps.getLength() != 0)
781  _data.waterplane_coeff = _data.waterplane_area / (ps.getBeam() * ps.getLength());
782  delete waterplane;
783  _data.km_transverse = _data.center_of_buoyancy.z() + _data.waterplane_mom_inertia.x() / _data.volume;
784  _data.km_longitudinal = _data.center_of_buoyancy.z() + _data.waterplane_mom_inertia.y() / _data.volume;
785  }
786 
787  // calculate block and prismatic coefficients
788  if (_draft != 0) {
789  if (_data.waterplane_area * _draft != 0)
790  _data.vert_prism_coefficient = _data.volume / (_data.waterplane_area * _draft);
791  if (ps.getHydrostaticCoefficients() == fcActualData) {
792  // block coefficient based on length and beam measured on waterline
793  if (submerged_width * submerged_length * _draft != 0)
794  _data.block_coefficient = _data.volume / (submerged_width * submerged_length * _draft);
795  // prismatic coefficient based on length and beam measured on waterline
796  if (_data.mainframe_area * submerged_length != 0)
797  _data.prism_coefficient = _data.volume / (_data.mainframe_area * submerged_length);
798  } else {
799  // block coefficient based on length and beam from settings
800  if (ps.getLength() * ps.getBeam() * _draft != 0)
801  _data.block_coefficient = _data.volume / (ps.getLength() * ps.getBeam() * _draft);
802  // prismatic coefficient based on length and beam from settings
803  if (_data.mainframe_area * ps.getLength() != 0)
804  _data.prism_coefficient = _data.volume / (_data.mainframe_area * ps.getLength());
805  }
806  }
807 
808  // calculate sectional areas
809  if (hasCalculation(hcSAC) || hasCalculation(hcAll)) {
810  if (_owner->getStations().size()) {
811  StationAreaCalculation sac(_data, _owner);
812  for_each(_owner->getStations().begin(), _owner->getStations().end(), sac);
813  }
814  }
815 
816  // calculate lateral area and center of gravity
817  if (hasCalculation(hcLateralArea) || hasCalculation(hcAll)) {
818  Intersection* lateralplane = new Intersection(_owner, fiButtock, Plane(0,1,0,0.001), true);
819  lateralplane->calculateArea(_data.waterline_plane, &_data.lateral_area, &_data.lateral_cog, &tmpp2d);
820  delete lateralplane;
821  }
822 
823  // all done!
824  setCalculated(true);
825 }
