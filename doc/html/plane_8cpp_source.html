49 Plane::Plane(const QVector3D& p1, const QVector3D& p2, const QVector3D& p3)
50 {
51  // calculate normal
52  QVector3D n;
53  n.setX((p2.y() - p1.y()) * (p3.x() - p1.x()) - (p2.x() - p1.x()) * (p3.y() - p1.y()));
54  n.setY((p2.x() - p1.x()) * (p3.x() - p1.x()) - (p2.x() - p1.x()) * (p3.x() - p1.x()));
55  n.setZ((p2.x() - p1.x()) * (p3.y() - p1.y()) - (p2.y() - p1.y()) * (p3.x() - p1.x()));
56  n.normalize();
57  _vars[0] = n.x() / n.length();
58  _vars[1] = n.y() / n.length();
59  _vars[2] = n.z() / n.length();
60  _vars[3] = -p1.x() * _vars[0] - p1.y() * _vars[1] - p1.z() * _vars[2];
61 }
