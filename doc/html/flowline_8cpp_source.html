106 void Flowline::loadBinary(FileBuffer& source)
107 {
108  quint32 n;
109  float f;
110  bool b;
111  QVector3D p;
112  source.load(f);
113  _projection_point.setX(f);
114  source.load(f);
115  _projection_point.setY(f);
116  source.load(n);
117  _projection_vw = static_cast<viewport_type_t>(n);
118  source.load(_build);
119  source.load(b);
120  if (b)
121  _owner->setSelectedFlowline(this);
122  source.load(n);
123  for (size_t i=0; i<n; i++) {
124  source.load(p);
125  _flowline->add(p);
126  source.load(b);
127  _flowline->setKnuckle(i, b);
128  }
129 }
