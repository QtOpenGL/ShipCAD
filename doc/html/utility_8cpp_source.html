121 int ShipCAD::FindDXFColorIndex(QColor color)
122 {
123  int result = 1;
124  int r = color.red();
125  int g = color.green();
126  int b = color.blue();
127  float dist = 1E30f;
128  for (int i=0; i<255; ++i) {
129  int r1 = DXFLayerColors[i].red();
130  int g1 = DXFLayerColors[i].green();
131  int b1 = DXFLayerColors[i].blue();
132  float tmp = sqrt(static_cast<float>((r1-r)*(r1-r) + (g1-g)*(g1-g) + (b1-b)*(b1-b)));
133  if (tmp < dist) {
134  dist = tmp;
135  result = i+1;
136  }
137  }
138  return result;
139 }

172 static bool SameSide(const QVector3D& p1,
173  const QVector3D& p2,
174  const QVector3D& a,
175  const QVector3D& b)
176 {
177  QVector3D ba = b - a;
178  QVector3D p = p1 - a;
179  QVector3D cp1 = QVector3D::crossProduct(p, ba);
180  p = p2 - a;
181  QVector3D cp2 = QVector3D::crossProduct(p, ba);
182  float dp = QVector3D::dotProduct(cp1, cp2);
183  if (fabs(dp) < 1E5f)
184  dp = dp - 1 + 1;
185  return dp >= 0;
186 }

197 // clip a triangle given the 3 distances from a plane,
198 // returns 2 sets of coordinates, front of the plane, and back of the plane
199 void ShipCAD::ClipTriangle(const QVector3D& p1,
200  const QVector3D& p2,
201  const QVector3D& p3,
202  float s1,
203  float s2,
204  float s3,
205  std::vector<QVector3D>& front,
206  std::vector<QVector3D>& back)
207 {
208  if (s1 <=0 && s2 <= 0 && s3 <= 0) {
209  // all at the back of the plane
210  back.push_back(p1);
211  back.push_back(p2);
212  back.push_back(p3);
213  }
214  else if (s1 > 0 && s2 > 0 && s3 > 0) {
215  // all at the front of the plane
216  front.push_back(p1);
217  front.push_back(p2);
218  front.push_back(p3);
219  }
220  else {
221  // triangle spans the plane, calculate the intersections
222  if (s1 <= 0)
223  back.push_back(p1);
224  if (s1 >= 0)
225  front.push_back(p1);
226  if ((s1 < 0 && s2 > 0) || (s1 > 0 && s2 < 0)) {
227  float t;
228  if (s1 == s2)
229  t = 0.5;
230  else
231  t = -s1 / (s2 - s1);
232  QVector3D p = p1 + (t * (p2 - p1));
233  back.push_back(p);
234  front.push_back(p);
235  }
236  if (s2 <= 0)
237  back.push_back(p2);
238  if (s2 >= 0)
239  front.push_back(p2);
240  if ((s2 < 0 && s3 > 0) || (s2 > 0 && s3 < 0)) {
241  float t;
242  if (s2 == s3)
243  t = 0.5;
244  else
245  t = -s2 / (s3 - s2);
246  QVector3D p = p2 + (t * (p3 - p2));
247  back.push_back(p);
248  front.push_back(p);
249  }
250  if (s3 <= 0)
251  back.push_back(p3);
252  if (s3 >= 0)
253  front.push_back(p3);
254  if ((s3 < 0 && s1 > 0) || (s3 > 0 && s1 < 0)) {
255  float t;
256  if (s2 == s3)
257  t = 0.5;
258  else
259  t = -s3 / (s1 - s3);
260  QVector3D p = p3 + (t * (p1 - p3));
261  back.push_back(p);
262  front.push_back(p);
263  }
264  }
265 }

364  else if (min == d4)
365  fixed->insert_spline(fixed->numberOfPoints(), true, true, *match);
366  else
367  throw runtime_error("Error in comparing minimum values JoinSplineSegments");
368  // destroy the matching spline
369  list.erase(list.begin()+matchindex-1);
370  delete match;
371  i = 0; // reset match index to start searching for a new matching spline
372  }
373  }
374  }
375  ++i;
376  }
