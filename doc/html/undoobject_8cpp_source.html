52 void UndoObject::accept()
53 {
54  if (_owner->undoCount() > 0) {
55  UndoObject* last = _owner->getUndoObject(_owner->undoCount() - 1);
56  if (last->isTempRedoObj()) {
57  _owner->deleteUndoObject(last);
58  delete last;
59  }
60  }
61  // delete all undo objects after the current one
62  for (size_t i=_owner->undoCount(); i>_owner->undoPosition()+1; i--) {
63  UndoObject* last = _owner->getUndoObject(i-1);
64  _owner->deleteUndoObject(last);
65  delete last;
66  }
67  _owner->addUndoObject(this);
68  _owner->setUndoPosition(_owner->undoCount());
69  // remove objects from the front of the list until memory is within limits or we have 2 items
70  while (_owner->undoCount() > 2 &&
71  (_owner->getUndoMemory() / (1024*1024)) > _owner->getPreferences().getMaxUndoMemory()) {
72  UndoObject* first = _owner->getUndoObject(0);
73  _owner->deleteUndoObject(first);
74  delete first;
75  _owner->setUndoPosition(_owner->undoPosition()-1);
76  _owner->setPrevUndoPosition(_owner->prevUndoPosition()-1);
77  }
78 }
