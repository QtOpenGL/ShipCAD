93 void SubdivisionLayer::processTriangle(const QVector3D& p1,
94  const QVector3D& p2,
95  const QVector3D& p3,
96  LayerProperties& props)
97 {
98  QVector3D center = (p1 + p2 + p3) / 3.0f;
99  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z())
100  + (p2.y() - p3.y()) * (p2.z() + p3.z())
101  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
102  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x())
103  + (p2.z() - p3.z()) * (p2.x() + p3.x())
104  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
105  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y())
106  + (p2.x() - p3.x()) * (p2.y() + p3.y())
107  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
108  float area = sqrt(ax * ax + ay * ay + az * az);
109  props.surface_area += area;
110  props.surface_center_of_gravity += (area * center);
111 }

113 LayerProperties SubdivisionLayer::getSurfaceProperties()
114 {
115  LayerProperties result;
116  result.surface_area = 0;
117  result.weight = 0;
118  result.surface_center_of_gravity = ZERO;
119 
120  for (size_t i=1; i<=_patches.size(); ++i) {
121  for (size_t j=1; j<=_patches[i-1]->numberOfChildren(); ++j) {
122  SubdivisionFace* child = _patches[i-1]->getChild(j-1);
123  for (size_t k=3; k<=child->numberOfPoints(); ++k) {
124  processTriangle(child->getPoint(0)->getCoordinate(),
125  child->getPoint(k-2)->getCoordinate(),
126  child->getPoint(k-1)->getCoordinate(),
127  result);
128  }
129  }
130  }
131  if (result.surface_area != 0) {
132  result.surface_center_of_gravity /= result.surface_area;
133  if (isSymmetric()) {
134  result.surface_area *= 2.0f;
135  result.surface_center_of_gravity.setY(0);
136  }
137  result.weight = result.surface_area * _thickness * _material_density;
138  }
139  return result;
140 }

227 bool SubdivisionLayer::calculateIntersectionPoints(SubdivisionLayer* layer)
228 {
229  bool result = false;
230  vector<SubdivisionControlEdge*> edges;
231  vector<SubdivisionControlPoint*> newpoints;
232 
233  SubdivisionControlPoint* p1, *p2;
234  SubdivisionControlEdge* edge;
235  SubdivisionControlFace* face;
236 
237  // assemble all controledges in a list
238  for (size_t i=1; i<=_patches.size(); ++i) {
239  face = _patches[i-1];
240  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
241  for (size_t j=1; j<=face->numberOfPoints(); ++j) {
242  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j-1));
243  edge = _owner->controlEdgeExists(p1, p2);
244  if (edge != 0 && find(edges.begin(), edges.end(), edge) == edges.end())
245  edges.push_back(edge);
246  p1 = p2;
247  }
248  }
249 
250  // now check all edges for intersection with layer 2
251  size_t i = 1;
252  while (i <= edges.size()) {
253  edge = edges[i-1];
254  bool intfound = false;
255  size_t j = 1;
256  while (j <= layer->numberOfFaces() && !intfound) {
257  face = layer->getFace(j-1);
258  size_t k = 1;
259  while (k <= face->numberOfChildren() && !intfound) {
260  SubdivisionFace* child = face->getChild(k-1);
261  size_t l = 3;
262  while (l <= child->numberOfPoints() && !intfound) {
263  Plane pln(child->getPoint(0)->getCoordinate(),
264  child->getPoint(l-2)->getCoordinate(),
265  child->getPoint(l-1)->getCoordinate());
266  float s1 = pln.a() * edge->startPoint()->getCoordinate().x()
267  + pln.b() * edge->startPoint()->getCoordinate().y()
268  + pln.c() * edge->startPoint()->getCoordinate().z()
269  + pln.d();
270  float s2 = pln.a() * edge->endPoint()->getCoordinate().x()
271  + pln.b() * edge->endPoint()->getCoordinate().y()
272  + pln.c() * edge->endPoint()->getCoordinate().z()
273  + pln.d();
274  if ((s1 < 0 && s2 > 0) || (s1 > 0 && s2 < 0)) {
275  // edge intersects the plane, does it lie in the triangle?
276  float t;
277  if (s1 == s2)
278  t = 0.5;
279  else
280  t = -s1 / (s2 - s1);
281  QVector3D p3d = edge->startPoint()->getCoordinate()
282  + t * (edge->endPoint()->getCoordinate()
283  - edge->startPoint()->getCoordinate());
284  if (PointInTriangle(p3d, child->getPoint(0)->getCoordinate(),
285  child->getPoint(l-2)->getCoordinate(),
286  child->getPoint(l-1)->getCoordinate())) {
287  // yes, valid intersection
288  SubdivisionControlPoint* p = edge->insertControlPoint(p3d);
289  if (p != 0) {
290  intfound = true;
291  result = true;
292  newpoints.push_back(p);
293  }
294  }
295  }
296  l++;
297  }
298  k++;
299  }
300  j++;
301  }
302  i++;
303  }
304  if (newpoints.size() > 0) {
305  // try to find multiple new points belonging to the same face and insert an edge
306  size_t i = 1;
307  edges.clear();
308  bool inserted = false;
309  while (i <= newpoints.size()) {
310  p1 = newpoints[i-1];
311  size_t j = 1;
312  while (j <= p1->numberOfFaces()) {
313  face = dynamic_cast<SubdivisionControlFace*>(p1->getFace(j-1));
314  size_t k = 1;
315  while (k <= face->numberOfPoints() && !inserted) {
316  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(k-1));
317  if (p1 != p2 && find(newpoints.begin(), newpoints.end(), p2) != newpoints.end()) {
318  // this is also a new point, first check if an edge already exists between p1 and p2
319  if (_owner->edgeExists(p1, p2) == 0) {
320  inserted = true;
321  edge = face->insertControlEdge(p1, p2);
322  edge->setSelected(true);
323  edges.push_back(edge);
324  }
325  }
326  k++;
327  }
328  if (!inserted)
329  j++;
330  }
331  i++;
332  }
333  }
334  return result;
335 }

370 void SubdivisionLayer::draw(Viewport& vp)
371 {
372  if (!isVisible() || numberOfFaces() == 0)
373  return;
374  if (vp.getViewportMode() != vmWireFrame) {
375  // not vmWireFrame, but shaded
376  if (vp.getViewportMode() == vmShadeGauss) {
377  for (size_t i=0; i<numberOfFaces(); ++i)
378  getFace(i)->drawCurvatureFaces(vp,
379  _owner->getMinGausCurvature(),
380  _owner->getMaxGausCurvature());
381  }
382  else {
383 // FaceShader* faceshader = vp.setMonoFaceShader();
384  FaceShader* faceshader = vp.setLightedFaceShader();
385  for (size_t i=0; i<numberOfFaces(); ++i)
386  getFace(i)->drawFaces(vp, faceshader);
387  }
388  }
389  else {
390  LineShader* lineshader = vp.setLineShader();
391  // vmWireFrame
392  if (_owner->showInteriorEdges()) {
393  for (size_t i=0; i<numberOfFaces(); ++i)
394  getFace(i)->draw(vp, lineshader);
395  }
396  // draw all interior crease-edges
397  for (size_t i=0; i<numberOfFaces(); ++i) {
398  SubdivisionControlFace* face = getFace(i);
399  for (size_t j=0; j<face->numberOfControlEdges(); ++j) {
400  SubdivisionEdge* edge = face->getControlEdge(j);
401  if (edge->isCrease())
402  edge->draw(_owner->drawMirror() && isSymmetric(), vp, lineshader, _owner->getCreaseColor());
403  }
404  }
