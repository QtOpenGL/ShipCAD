96 void SubdivisionLayer::processTriangle(const QVector3D& p1,
97  const QVector3D& p2,
98  const QVector3D& p3,
99  LayerProperties& props)
100 {
101  QVector3D center = (p1 + p2 + p3) / 3.0f;
102  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z())
103  + (p2.y() - p3.y()) * (p2.z() + p3.z())
104  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
105  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x())
106  + (p2.z() - p3.z()) * (p2.x() + p3.x())
107  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
108  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y())
109  + (p2.x() - p3.x()) * (p2.y() + p3.y())
110  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
111  float area = sqrt(ax * ax + ay * ay + az * az);
112  props.surface_area += area;
113  props.surface_center_of_gravity += (area * center);
114 }

116 LayerProperties SubdivisionLayer::getSurfaceProperties()
117 {
118  LayerProperties result;
119  result.surface_area = 0;
120  result.weight = 0;
121  result.surface_center_of_gravity = ZERO;
122 
123  for (size_t i=1; i<=_patches.size(); ++i) {
124  for (size_t j=1; j<=_patches[i-1]->numberOfChildren(); ++j) {
125  SubdivisionFace* child = _patches[i-1]->getChild(j-1);
126  for (size_t k=3; k<=child->numberOfPoints(); ++k) {
127  processTriangle(child->getPoint(0)->getCoordinate(),
128  child->getPoint(k-2)->getCoordinate(),
129  child->getPoint(k-1)->getCoordinate(),
130  result);
131  }
132  }
133  }
134  if (result.surface_area != 0) {
135  result.surface_center_of_gravity /= result.surface_area;
136  if (isSymmetric()) {
137  result.surface_area *= 2.0f;
138  result.surface_center_of_gravity.setY(0);
139  }
140  result.weight = result.surface_area * _thickness * _material_density;
141  }
142  return result;
143 }

230 bool SubdivisionLayer::calculateIntersectionPoints(SubdivisionLayer* layer)
231 {
232  bool result = false;
233  vector<SubdivisionControlEdge*> edges;
234  vector<SubdivisionControlPoint*> newpoints;
235 
236  SubdivisionControlPoint* p1, *p2;
237  SubdivisionControlEdge* edge;
238  SubdivisionControlFace* face;
239 
240  // assemble all controledges in a list
241  for (size_t i=1; i<=_patches.size(); ++i) {
242  face = _patches[i-1];
243  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
244  for (size_t j=1; j<=face->numberOfPoints(); ++j) {
245  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j-1));
246  edge = _owner->controlEdgeExists(p1, p2);
247  if (edge != 0 && find(edges.begin(), edges.end(), edge) == edges.end())
248  edges.push_back(edge);
249  p1 = p2;
250  }
251  }
252 
253  // now check all edges for intersection with layer 2
254  size_t i = 1;
255  while (i <= edges.size()) {
256  edge = edges[i-1];
257  bool intfound = false;
258  size_t j = 1;
259  while (j <= layer->numberOfFaces() && !intfound) {
260  face = layer->getFace(j-1);
261  size_t k = 1;
262  while (k <= face->numberOfChildren() && !intfound) {
263  SubdivisionFace* child = face->getChild(k-1);
264  size_t l = 3;
265  while (l <= child->numberOfPoints() && !intfound) {
266  Plane pln(child->getPoint(0)->getCoordinate(),
267  child->getPoint(l-2)->getCoordinate(),
268  child->getPoint(l-1)->getCoordinate());
269  float s1 = pln.a() * edge->startPoint()->getCoordinate().x()
270  + pln.b() * edge->startPoint()->getCoordinate().y()
271  + pln.c() * edge->startPoint()->getCoordinate().z()
272  + pln.d();
273  float s2 = pln.a() * edge->endPoint()->getCoordinate().x()
274  + pln.b() * edge->endPoint()->getCoordinate().y()
275  + pln.c() * edge->endPoint()->getCoordinate().z()
276  + pln.d();
277  if ((s1 < 0 && s2 > 0) || (s1 > 0 && s2 < 0)) {
278  // edge intersects the plane, does it lie in the triangle?
279  float t;
280  if (s1 == s2)
281  t = 0.5;
282  else
283  t = -s1 / (s2 - s1);
284  QVector3D p3d = edge->startPoint()->getCoordinate()
285  + t * (edge->endPoint()->getCoordinate()
286  - edge->startPoint()->getCoordinate());
287  if (PointInTriangle(p3d, child->getPoint(0)->getCoordinate(),
288  child->getPoint(l-2)->getCoordinate(),
289  child->getPoint(l-1)->getCoordinate())) {
290  // yes, valid intersection
291  SubdivisionControlPoint* p = edge->insertControlPoint(p3d);
292  if (p != 0) {
293  intfound = true;
294  result = true;
295  newpoints.push_back(p);
296  }
297  }
298  }
299  l++;
300  }
301  k++;
302  }
303  j++;
304  }
305  i++;
306  }
307  if (newpoints.size() > 0) {
308  // try to find multiple new points belonging to the same face and insert an edge
309  size_t i = 1;
310  edges.clear();
311  bool inserted = false;
312  while (i <= newpoints.size()) {
313  p1 = newpoints[i-1];
314  size_t j = 1;
315  while (j <= p1->numberOfFaces()) {
316  face = dynamic_cast<SubdivisionControlFace*>(p1->getFace(j-1));
317  size_t k = 1;
318  while (k <= face->numberOfPoints() && !inserted) {
319  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(k-1));
320  if (p1 != p2 && find(newpoints.begin(), newpoints.end(), p2) != newpoints.end()) {
321  // this is also a new point, first check if an edge already exists between p1 and p2
322  if (_owner->edgeExists(p1, p2) == 0) {
323  inserted = true;
324  edge = face->insertControlEdge(p1, p2);
325  edge->setSelected(true);
326  edges.push_back(edge);
327  }
328  }
329  k++;
330  }
331  if (!inserted)
332  j++;
333  }
334  i++;
335  }
336  }
337  return result;
338 }

373 void SubdivisionLayer::draw(Viewport& vp)
374 {
375  if (!isVisible() || numberOfFaces() == 0)
376  return;
377  if (vp.getViewportMode() != Viewport::vmWireFrame) {
378  // not vmWireFrame, but shaded
379  if (vp.getViewportMode() == Viewport::vmShadeGauss) {
380  for (size_t i=0; i<numberOfFaces(); ++i)
381  getFace(i)->drawCurvatureFaces(vp,
382  _owner->getMinGausCurvature(),
383  _owner->getMaxGausCurvature());
384  }
385  else {
386  MonoFaceShader* monoshader = vp.setMonoFaceShader();
387  for (size_t i=0; i<numberOfFaces(); ++i)
388  getFace(i)->drawFaces(vp, monoshader);
389  }
390  }
391  else {
392  LineShader* lineshader = vp.setLineShader();
393  // vmWireFrame
394  if (_owner->showInteriorEdges()) {
395  for (size_t i=0; i<numberOfFaces(); ++i)
396  getFace(i)->draw(vp, lineshader);
397  }
398  // draw all interior crease-edges
399  for (size_t i=0; i<numberOfFaces(); ++i) {
400  SubdivisionControlFace* face = getFace(i);
401  for (size_t j=0; j<face->numberOfControlEdges(); ++j) {
402  SubdivisionEdge* edge = face->getControlEdge(j);
403  if (edge->isCrease())
404  edge->draw(_owner->drawMirror() && isSymmetric(), vp, lineshader, _owner->getCreaseColor());
405  }
406  }
407  }
