113 float SubdivisionPoint::Angle_VV_3D(const QVector3D& p1,
114  const QVector3D& p2,
115  const QVector3D& p3)
116 {
117  QVector3D v1 = p1 - p2;
118  QVector3D v2 = p3 - p2;
119  v1.normalize();
120  v2.normalize();
121  float l = v1.x()*v2.x() + v1.y()*v2.y() + v1.z()*v2.z();
122  if (l < -1)
123  l = -1;
124  else if (l > 1)
125  l = 1;
126  return acos(l);
127 }

129 float SubdivisionPoint::getCurvature()
130 {
131  float result = 0.0;
132  for (size_t i=0; i<_edges.size(); ++i)
133  if (_edges[i]->numberOfFaces() < 2)
134  return result;
135  float sigma = 0;
136  for (size_t j=0; j<_faces.size(); ++j) {
137  SubdivisionFace* face = _faces[j];
138  size_t index = face->indexOfPoint(this);
139  size_t previndex = (index + face->numberOfPoints() - 1) % face->numberOfPoints();
140  size_t nextindex = (previndex + 2) % face->numberOfPoints();
141  SubdivisionPoint* prev = face->getPoint(previndex);
142  SubdivisionPoint* next = face->getPoint(nextindex);
143  float tmp = Angle_VV_3D(prev->getCoordinate(),
144  _coordinate,
145  next->getCoordinate());
146  sigma += tmp;
147  }
148  result = (360 - RadToDeg(sigma));
149  return result;

171 bool SubdivisionPoint::isRegularPoint()
172 {
173  bool result = false;
174  // this procedure was only tested to TRACE regular quad edges
175  // and regular/irregular CREASE edges for dxf export
176  if (_faces.size() == 5 && _edges.size() == 5) {
177  // boundary of quad and triangle
178  int n = 0;
179  for (size_t i=0; i<_faces.size(); ++i) {
180  if (_faces[i]->numberOfPoints() == 3)
181  n++;
182  }
183  if (n == 3)
184  result = true;
185  }
186  else if (_faces.size() == 6 && _edges.size() == 6) {
187  // regular point with all triangles
188  int n = 0;
189  for (size_t i=0; i<_faces.size(); ++i) {
190  if (_faces[i]->numberOfPoints() == 3)
191  n++;
192  }
193  if (n == 6)
194  result = true;
195  }
196  else if (_faces.size() == 4 && _edges.size() == 4) {
197  // regular point with all quads
198  result = true;
199  }
200  else {
201  // regular quad boundary edge
202  int n = 0;
203  for (size_t i=0; i<_edges.size(); ++i) {
204  if (_edges[i]->numberOfFaces() == 1)
205  n++;
206  }
207  if (n == 2 && _edges.size() == 3)
208  result = true;
209  }

213 QVector3D SubdivisionPoint::getLimitPoint()
214 {
215  QVector3D result = ZERO;
216  SubdivisionPoint* p30 = 0;
217  SubdivisionPoint* p33 = 0;
218  SubdivisionPoint* p = 0;
219  if (_vtype == svDart || _vtype == svRegular) {
220  result = ZERO;
221  int n = _faces.size();
222  for (int i=0; i<n; ++i) {
223  SubdivisionFace* face = _faces[i];
224  size_t ind = face->indexOfPoint(this);
225  p30 = face->getPoint((ind + 1) % face->numberOfPoints());
226  p33 = face->getPoint((ind + 2) % face->numberOfPoints());
227  result += (n * _coordinate + 4 * p30->getCoordinate() + p33->getCoordinate());
228  }
229  result /= (n * (n + 5));
230  }
231  else if (_vtype == svCrease) {
232  for (size_t i=0; i<_edges.size(); ++i) {
233  SubdivisionEdge* edge = _edges[i];
234  if (edge->isCrease()) {
235  if (edge->startPoint() == this)
236  p = edge->endPoint();
237  else
238  p = edge->startPoint();
239  if (p30 == 0)
240  p30 = p;
241  else
242  p33 = p;
243  }
244  }
245  if (p30 != 0 && p33 != 0) {
246  result = (1/6.0f)*p30->getCoordinate()
247  + (2/3.0f)*result
248  + (1/6.0f)*p33->getCoordinate();
249  }
250  else {
251  // this is an error
252  result = getCoordinate();
253  }
254  }
255  else
256  result = getCoordinate();
257  return result;
258 }

308 QVector3D SubdivisionPoint::averaging()
309 {
310  QVector3D result;
311  SubdivisionPoint* p;
312  float totalweight = 0.0;
313  size_t nt = 0;
314  float weight;
315 
316  if (_edges.size() == 0 || _vtype == svCorner)
317  result = getCoordinate();
318  else {
319  if (_vtype == svCrease) {
320  result = getCoordinate() * 0.5;
321  for (size_t i=0; i<_edges.size(); ++i) {
322  SubdivisionEdge* edge = _edges[i];
323  if (edge->numberOfFaces() == 1 || edge->isCrease()) {
324  if (edge->startPoint() == this)
325  p = edge->endPoint();
326  else
327  p = edge->startPoint();
328  result += (0.25 * p->getCoordinate());
329  }
330  }
331  }
332  else {
333  result = ZERO;
334  for (size_t i=0; i<_faces.size(); ++i) {
335  SubdivisionFace* face = _faces[i];
336  QVector3D center = ZERO;
337  if (face->numberOfPoints() == 3) {
338  ++nt;
339  // calculate centerpoint
340  for (size_t j=0; j<face->numberOfPoints(); ++j) {
341  p = face->getPoint(j);
342  if (p == this)
343  weight = .25;
344  else
345  weight = .375;
346  center += (weight * p->getCoordinate());
347  }
348  weight = third_pi;
349  }
350  else if (face->numberOfPoints() == 4) {
351  for (size_t j=0; j<face->numberOfPoints(); ++j) {
352  p = face->getPoint(j);
353  weight = .25;
354  center += (weight * p->getCoordinate());
355  }
356  weight = half_pi;
357  }
358  else
359  throw runtime_error("invalid number of points in SubdivisionPoint::averaging");
360  result += (weight * center);
361  totalweight += weight;
362  }
363  if (totalweight != 0) {
364  result /= totalweight;
365  }
366  size_t nq = _faces.size() - nt;
367  float a;
368  if (nt == _faces.size()) {
369  // apply averaging in case of vertex surrounded by triangles
370  a = 5/3.0 - 8/3.0*sqrt(.375+.25*cos(two_pi/_faces.size()));
371  }
372  else if (nq == _faces.size()) {
373  // apply averaging in case of vertex surrounded by quads
374  a = 4 / static_cast<float>(_faces.size());
375  }
376  else {
377  // apply averaging in case of vertex on boundary of quads and triangles
378  if (nq == 0 && nt == 3)
379  a = 1.5;
380  else
381  a = 12 / static_cast<float>(3 * nq + 2 * nt);
382  }
383  if (a != 1.0) {
384  result = getCoordinate() + a * (result - getCoordinate());
385  }
386  }
387  }
388  return result;
389 }

438 QVector3D SubdivisionPoint::getNormal()
439 {
440  QVector3D result;
441  for (size_t i=0; i<_faces.size(); ++i) {
442  SubdivisionFace* face = _faces[i];
443  if (face->numberOfPoints() > 4) {
444  // face possibly concave at this point
445  // use the normal of all points from this face
446  QVector3D c = face->getFaceCenter();
447  QVector3D n = ZERO;
448  for (size_t j=1; j<face->numberOfPoints(); ++j) {
449  n = n + UnifiedNormal(c, face->getPoint(j-1)->getCoordinate(),
450  face->getPoint(j)->getCoordinate());
451  }
452  n.normalize();
453  result = result + n;
454  }
455  else {
456  size_t index = face->indexOfPoint(this);
457  size_t j = (index + face->numberOfPoints() - 1) % face->numberOfPoints();
458  QVector3D p1 = face->getPoint(j)->getCoordinate();
459  j = (index + face->numberOfPoints() + 1) % face->numberOfPoints();
460  QVector3D p3 = face->getPoint(j)->getCoordinate();
461  result = result + UnifiedNormal(p1, getCoordinate(), p3);
462  }
463  }
464  result.normalize();
465  return result;
466 }

619 void SubdivisionControlPoint::collapse()
620 {
621  SubdivisionControlPoint* p1, *p2;
622  SubdivisionControlEdge* edge1, *edge2;
623  SubdivisionControlFace* face;
624  if (_faces.size() <= 2) {
625  setSelected(false);
626  p1 = 0;
627  p2 = 0;
628  // this is possibly a point on a boundary edge
629  // check for this special case
630  edge1 = 0;
631  edge2 = 0;
632  for (size_t i=0; i<numberOfEdges(); ++i) {
633  if (_edges[i]->numberOfFaces() == 1) {
634  if (edge1 == 0) {
635  edge1 = dynamic_cast<SubdivisionControlEdge*>(_edges[i]);
636  }
637  else {
638  edge2 = dynamic_cast<SubdivisionControlEdge*>(_edges[i]);
639  }
640  }
641  }
642  if (edge1 != 0 && edge2 != 0) {
643  for (size_t i=numberOfEdges(); i>0; --i) {
644  edge1 = dynamic_cast<SubdivisionControlEdge*>(_edges[i-1]);
645  _owner->collapseEdge(edge1);
646  }
647  }
648  bool edge_collapse = false;
649  bool crease = false;
650  if (_edges.size() == 2) {
651  edge_collapse = true;
652  edge1 = dynamic_cast<SubdivisionControlEdge*>(_edges[0]);
653  if (edge1 == 0)
654  throw runtime_error("edge is not a control edge SubdivisionControlPoint::collapse");
655  if (edge1->startPoint() == this) {
656  p1 = dynamic_cast<SubdivisionControlPoint*>(edge1->endPoint());
657  }
658  else {
659  p1 = dynamic_cast<SubdivisionControlPoint*>(edge1->startPoint());
660  }
661  edge2 = dynamic_cast<SubdivisionControlEdge*>(_edges[1]);
662  if (edge2 == 0)
663  throw runtime_error("edge is not a control edge SubdivisionControlPoint::collapse");
664  if (edge2->startPoint() == this) {
665  p2 = dynamic_cast<SubdivisionControlPoint*>(edge2->endPoint());
666  }
667  else {
668  p2 = dynamic_cast<SubdivisionControlPoint*>(edge2->startPoint());
669  }
670  crease = edge1->isCrease() || edge2->isCrease();
671  }
672  for (size_t i=numberOfFaces(); i>0; --i) {
673  face = dynamic_cast<SubdivisionControlFace*>(_faces[i-1]);
674  vector<SubdivisionControlPoint*> points;
675  for (size_t j=0; j<face->numberOfPoints(); ++j) {
676  SubdivisionControlPoint* pt = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j));
677  if (pt && pt != this)
678  points.push_back(pt);
679  }
680  _owner->addControlFace(points, false, face->getLayer());
681  _owner->deleteControlFace(face);
682  }
683 
684  if (edge_collapse) {
685  edge1 = _owner->controlEdgeExists(p1, p2);
686  if (edge1 != 0)
687  edge1->setCrease(crease);
688  }
689  } // faces.size <= 2
690  else {
691  vector<SubdivisionControlPoint*> checklist;
692  vector<SubdivisionControlEdge*> edges;
693  for (size_t i=0; i<numberOfEdges(); ++i) {
694  SubdivisionControlPoint* s = dynamic_cast<SubdivisionControlPoint*>(edges[i]->startPoint());
695  SubdivisionControlPoint* e = dynamic_cast<SubdivisionControlPoint*>(edges[i]->endPoint());
696  if (s == 0 || e == 0)
697  throw runtime_error("point not control point SubdivisionControlPoint::collapse");
698  if (s == this)
699  checklist.push_back(e);
700  else
701  checklist.push_back(s);
702  }
703  for (size_t i=0; i<numberOfFaces(); ++i) {
704  face = dynamic_cast<SubdivisionControlFace*>(_faces[i]);
705  if (face == 0)
706  throw runtime_error("face is not a control face SubdivisionControlPoint::collapse");
707  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
708  if (p1 == 0)
709  throw runtime_error("point not control point SubdivisionControlPoint::collapse");
710  for (size_t j=0; j<face->numberOfPoints(); ++j) {
711  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j));
712  if (p2 == 0)
713  throw runtime_error("point not control point SubdivisionControlPoint::collapse");
714  if (p1 != this && p2 != this) {
715  edge1 = _owner->controlEdgeExists(p1, p2);
716  if (edge1 != 0
717  && find(edges.begin(), edges.end(), edge1) == edges.end())
718  _edges.push_back(edge1);
719  }
720  p1 = p2;
721  }
722  }
723  // sort edges in correct order and add new face
724  if (edges.size() > 2) {
725  vector<vector<SubdivisionControlPoint*> > sorted;
726  _owner->isolateEdges(edges, sorted);
727  for (size_t i=0; i<sorted.size(); ++i) {
728  vector<SubdivisionControlPoint*>& points = sorted[i];
729  if (points.size() > 2) {
730  face = _owner->addControlFace(points, false);
731  if (face != 0) {
732  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
733  if (p1 == 0)
734  throw runtime_error("point not control point SubdivisionControlPoint::collapse");
735  for (size_t j=0; j<face->numberOfPoints(); j++) {
736  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j));
737  if (p2 == 0)
738  throw runtime_error("point not control point SubdivisionControlPoint::collapse");
739  // BUGBUG: doesn't do anything here
740  edge1 = _owner->controlEdgeExists(p1, p2);
741  p1 = p2;
742  }
743  }
744  }
745  }
746  }
747  // BUGBUG: supposed to delete point here
748  for (size_t i=checklist.size(); i>0; --i) {
749  p1 = checklist[i-1];
750  if (p1->numberOfFaces() > 1 && p1->numberOfEdges() == 2)
751  p1->collapse();
752  }

847  case svCorner:
848  cornerPoints << pt->getCoordinate();
849  break;
850  }
851  }
852  }
853  glPointSize(surface->getControlPointSize()+4.0);
854  if (selPoints.size() > 0)

856  glPointSize(surface->getControlPointSize());

869  }
