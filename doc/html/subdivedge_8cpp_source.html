113 void SubdivisionEdge::setCrease(bool val)
114 {
115  if (_faces.size() == 1)
116  val = true; // boundary edges must always be crease edges
117  if (val == _crease)
118  return;
119  _crease = val;
120  int n = 0;
121  SubdivisionPoint* startPoint = _points[0];
122  SubdivisionPoint* endPoint = _points[1];
123  for (size_t i = 0; i<startPoint->numberOfEdges(); ++i) {
124  SubdivisionEdge* edge = startPoint->getEdge(i);
125  if (edge->isCrease())
126  n++;
127  }
128  if (startPoint->getVertexType() == svCorner) {
129  if (startPoint->numberOfFaces() > 1 && n == 2)
130  startPoint->setVertexType(svCrease);
131  }
132  else {
133  if (n == 0)
134  startPoint->setVertexType(svRegular);
135  else if (n == 1)
136  startPoint->setVertexType(svDart);
137  else if (n == 2)
138  startPoint->setVertexType(svCrease);
139  else if (n > 2)
140  startPoint->setVertexType(svCorner);
141  }
142  n = 0;
143  for (size_t i=0; i<endPoint->numberOfEdges(); ++i) {
144  SubdivisionEdge* edge = endPoint->getEdge(i);
145  if (edge->isCrease())
146  n++;
147  }
148  if (endPoint->getVertexType() == svCorner) {
149  if (endPoint->numberOfFaces() > 1 && n == 2)
150  endPoint->setVertexType(svCrease);
151  }
152  else {
153  if (n == 0)
154  endPoint->setVertexType(svRegular);
155  else if (n == 1)
156  endPoint->setVertexType(svDart);
157  else if (n == 2)
158  endPoint->setVertexType(svCrease);
159  else if (n > 2)
160  endPoint->setVertexType(svCorner);
161  }
162  // BUGBUG: this fails during subdivide
163  //_owner->setBuild(false);
164 }

166 SubdivisionEdge* SubdivisionEdge::getPreviousEdge()
167 {
168  SubdivisionPoint* p = _points[0];
169  SubdivisionEdge* result = 0;
170  if (p->isRegularPoint() && p->getVertexType() != svCorner) {
171  // find previous edge
172  for (size_t i=0; i<p->numberOfEdges(); ++i) {
173  if (p->getEdge(i) == this)
174  continue;
175  SubdivisionEdge* edge = p->getEdge(i);
176  if (edge->isCrease() == isCrease()) {
177  bool shares_face = false;
178  for (size_t j=0; j<_faces.size(); ++j) {
179  if (edge->hasFace(_faces[j])) {
180  shares_face = true;
181  break;
182  }
183  }
184  if (!shares_face) {
185  if (edge->startPoint() == startPoint())
186  edge->swapData();
187  result = edge;
188  break;
189  }
190  }
191  }
192  }
193  return result;
194 }

261 void SubdivisionEdge::draw(bool draw_mirror, Viewport& vp, LineShader* lineshader, const QColor& edgeColor)
262 {
263  QVector3D p1 = startPoint()->getCoordinate();
264  QVector3D p2 = endPoint()->getCoordinate();
265  QVector<QVector3D> vertices;
266  if (!draw_mirror && vp.getViewportType() == fvBodyplan) {
267  float mfl = _owner->getMainframeLocation();
268  if ((p1.x() < mfl && p2.x() > mfl) || (p1.x() > mfl && p2.x() < mfl)) {
269  // straddles mainframe
270  QVector3D m;
271  if (p2.x() - p1.x() != 0)
272  m = Interpolate(p1, p2, (mfl - p1.x()) / (p2.x() - p1.x()));
273  else
274  m = MidPoint(p1, p2);
275  if (p1.x() <= mfl) {
276  // p2 lies on port
277  vertices << p2;
278  vertices << m;
279  // p1 lies on stbd
280  p1.setY(-p1.y());
281  m.setY(-m.y());
282  vertices << p1;
283  vertices << m;
284  }
285  else {
286  // p2 lies on port
287  vertices << p1;
288  vertices << m;
289  // p2 lies on stbd
290  p2.setY(-p2.y());
291  m.setY(-m.y());
292  vertices << p2;
293  vertices << m;
294  }
295  }
296  else {
297  if (p1.x() <= mfl)
298  p1.setY(-p1.y());
299  if (p2.x() <= mfl)
300  p2.setY(-p2.y());
301  vertices << p1;
302  vertices << p2;
303  }
304  }
305  else {
306  // draw_mirror or not a bodyplan
307  vertices << p1;
308  vertices << p2;
309  if (draw_mirror) {
310  p1.setY(p1.y());
311  p2.setY(p2.y());
312  vertices << p1;
313  vertices << p2;
314  }
315  }
316  lineshader->renderLines(vertices, edgeColor);

482 SubdivisionControlPoint* SubdivisionControlEdge::insertControlPoint(const QVector3D& p)
483 {
484  SubdivisionControlPoint* result = SubdivisionControlPoint::construct(_owner);
485  SubdivisionControlPoint* sp = dynamic_cast<SubdivisionControlPoint*>(startPoint());
486  SubdivisionControlPoint* ep = dynamic_cast<SubdivisionControlPoint*>(endPoint());
487  if (sp == 0 || ep == 0)
488  throw runtime_error("start and/or end point are not control points");
489  result->setCoordinate(p);
490  if (getCurve() != 0) {
491  // insert the new point in the controlcurve
492  getCurve()->insertControlPoint(sp, ep, result);
493  }
494  _owner->addControlPoint(result);
495  for (size_t i=0; i<_faces.size(); ++i) {
496  SubdivisionFace* face = _faces[i];
497  if (face->hasPoint(sp) && face->hasPoint(ep)) {
498  size_t i1 = face->indexOfPoint(sp);
499  size_t i2 = face->indexOfPoint(ep);
500  if (i2 == i1+1)
501  face->insertPoint(i2, result);
502  else if (i1 == i2+1)
503  face->insertPoint(i1, result);
504  else if (i1 == 0 && i2 == face->numberOfPoints() - 1)
505  face->insertPoint(0, result);
506  else if (i2 == 0 && i1 == face->numberOfPoints() - 1)
507  face->insertPoint(0, result);
508  result->addFace(face);
509  }
510  }
511  ep->deleteEdge(this);
512  SubdivisionControlEdge* edge = _owner->addControlEdge(result, ep);
513  edge->setCrease(isCrease());
514  edge->setCurve(getCurve());
515  if (getCurve())
516  result->setVertexType(svCrease);
517  for (size_t i=0; i<_faces.size(); ++i)
518  edge->addFace(_faces[i]);
519  _points[1] = result;
520  result->addEdge(this);
521  return result;
522 }
