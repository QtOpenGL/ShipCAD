114 void SubdivisionEdge::setCrease(bool val)
115 {
116  if (_faces.size() == 1)
117  val = true; // boundary edges must always be crease edges
118  if (val == _crease)
119  return;
120  _crease = val;
121  int n = 0;
122  SubdivisionPoint* startPoint = _points[0];
123  SubdivisionPoint* endPoint = _points[1];
124  for (size_t i = 0; i<startPoint->numberOfEdges(); ++i) {
125  SubdivisionEdge* edge = startPoint->getEdge(i);
126  if (edge->isCrease())
127  n++;
128  }
129  if (startPoint->getVertexType() == SubdivisionPoint::svCorner) {
130  if (startPoint->numberOfFaces() > 1 && n == 2)
131  startPoint->setVertexType(SubdivisionPoint::svCrease);
132  }
133  else {
134  if (n == 0)
135  startPoint->setVertexType(SubdivisionPoint::svRegular);
136  else if (n == 1)
137  startPoint->setVertexType(SubdivisionPoint::svDart);
138  else if (n == 2)
139  startPoint->setVertexType(SubdivisionPoint::svCrease);
140  else if (n > 2)
141  startPoint->setVertexType(SubdivisionPoint::svCorner);
142  }
143  n = 0;
144  for (size_t i=0; i<endPoint->numberOfEdges(); ++i) {
145  SubdivisionEdge* edge = endPoint->getEdge(i);
146  if (edge->isCrease())
147  n++;
148  }
149  if (endPoint->getVertexType() == SubdivisionPoint::svCorner) {
150  if (endPoint->numberOfFaces() > 1 && n == 2)
151  endPoint->setVertexType(SubdivisionPoint::svCrease);
152  }
153  else {
154  if (n == 0)
155  endPoint->setVertexType(SubdivisionPoint::svRegular);
156  else if (n == 1)
157  endPoint->setVertexType(SubdivisionPoint::svDart);
158  else if (n == 2)
159  endPoint->setVertexType(SubdivisionPoint::svCrease);
160  else if (n > 2)
161  endPoint->setVertexType(SubdivisionPoint::svCorner);
162  }
163  // BUGBUG: this fails during subdivide
164  //_owner->setBuild(false);
165 }

167 SubdivisionEdge* SubdivisionEdge::getPreviousEdge()
168 {
169  SubdivisionPoint* p = _points[0];
170  SubdivisionEdge* result = 0;
171  if (p->isRegularPoint() && p->getVertexType() != SubdivisionPoint::svCorner) {
172  // find previous edge
173  for (size_t i=0; i<p->numberOfEdges(); ++i) {
174  if (p->getEdge(i) == this)
175  continue;
176  SubdivisionEdge* edge = p->getEdge(i);
177  if (edge->isCrease() == isCrease()) {
178  bool shares_face = false;
179  for (size_t j=0; j<_faces.size(); ++j) {
180  if (edge->hasFace(_faces[j])) {
181  shares_face = true;
182  break;
183  }
184  }
185  if (!shares_face) {
186  if (edge->startPoint() == startPoint())
187  edge->swapData();
188  result = edge;
189  break;
190  }
191  }
192  }
193  }
194  return result;
195 }

262 void SubdivisionEdge::draw(bool draw_mirror, Viewport& vp, LineShader* lineshader, const QColor& edgeColor)
263 {
264  QVector3D p1 = startPoint()->getCoordinate();
265  QVector3D p2 = endPoint()->getCoordinate();
266  QVector<QVector3D> vertices;
267  if (!draw_mirror && vp.getViewportType() == Viewport::fvBodyplan) {
268  float mfl = _owner->getMainframeLocation();
269  if ((p1.x() < mfl && p2.x() > mfl) || (p1.x() > mfl && p2.x() < mfl)) {
270  // straddles mainframe
271  QVector3D m;
272  if (p2.x() - p1.x() != 0)
273  m = Interpolate(p1, p2, (mfl - p1.x()) / (p2.x() - p1.x()));
274  else
275  m = MidPoint(p1, p2);
276  if (p1.x() <= mfl) {
277  // p2 lies on port
278  vertices << p2;
279  vertices << m;
280  // p1 lies on stbd
281  p1.setY(-p1.y());
282  m.setY(-m.y());
283  vertices << p1;
284  vertices << m;
285  }
286  else {
287  // p2 lies on port
288  vertices << p1;
289  vertices << m;
290  // p2 lies on stbd
291  p2.setY(-p2.y());
292  m.setY(-m.y());
293  vertices << p2;
294  vertices << m;
295  }
296  }
297  else {
298  if (p1.x() <= mfl)
299  p1.setY(-p1.y());
300  if (p2.x() <= mfl)
301  p2.setY(-p2.y());
302  vertices << p1;
303  vertices << p2;
304  }
305  }
306  else {
307  // draw_mirror or not a bodyplan
308  vertices << p1;
309  vertices << p2;
310  if (draw_mirror) {
311  p1.setY(p1.y());
312  p2.setY(p2.y());
313  vertices << p1;
314  vertices << p2;
315  }
316  }
317  lineshader->renderLines(vertices, edgeColor);
318 }

408 void SubdivisionControlEdge::collapse()
409 {
410  const type_info& this_type = typeid(this);
411  if (_faces.size() == 2 \
412  && typeid(_faces[0]) == this_type
413  && typeid(_faces[1]) == this_type) {
414 
415  SubdivisionControlPoint* s,*e;
416  SubdivisionControlFace* face1,*face2;
417  SubdivisionPoint* p1, *p2;
418 
419  if (startPoint()->numberOfEdges() > 2 && endPoint()->numberOfEdges() > 2) {
420  if (getCurve() != 0)
421  getCurve()->deleteEdge(this);
422  if (isSelected())
423  setSelected(false);
424  s = dynamic_cast<SubdivisionControlPoint*>(startPoint());
425  e = dynamic_cast<SubdivisionControlPoint*>(endPoint());
426  _owner->setBuild(false);
427  face1 = dynamic_cast<SubdivisionControlFace*>(_faces[0]);
428  face2 = dynamic_cast<SubdivisionControlFace*>(_faces[1]);
429  // check faces for consistent ordering of the points (same normal direction)
430  // because inconsistent ordering can lead to access violations
431  p1 = face1->getPoint(face1->numberOfPoints()-1);
432  for (size_t i=0; i<face1->numberOfPoints(); ++i) {
433  p2 = face1->getPoint(i);
434  if ((p1 == startPoint() && p2 == endPoint())
435  || ((p2 == startPoint() && p1 == endPoint()))) {
436  size_t ind1 = face2->indexOfPoint(p2);
437  size_t ind2 = (ind1 + 1) % face2->numberOfPoints(); // select the next index
438  if (face2->getPoint(ind2) != p1) {
439  face2->flipNormal();
440  }
441  break;
442  }
443  else
444  p1 = p2;
445  }
446  }
447 
448  SubdivisionLayer* layer = face1->getLayer();
449  // remove the control faces from the layers they belong to
450  face1->getLayer()->deleteControlFace(face1);
451  face2->getLayer()->deleteControlFace(face2);
452  size_t ind1 = face1->indexOfPoint(startPoint());
453  size_t ind2 = face1->indexOfPoint(endPoint());
454  if (ind2 < ind1 && fabs(static_cast<float>(ind2 - ind1)) == 1.0f)
455  swap(ind1, ind2);
456  size_t ind3 = face2->indexOfPoint(startPoint());
457  size_t ind4 = face2->indexOfPoint(endPoint());
458  if (ind4 < ind3 && fabs(static_cast<float>(ind4 - ind3)) == 1.0f)
459  swap(ind3, ind4);
460  if (ind1 == 0 && ind2 == face1->numberOfPoints() - 1
461  && ind3 == 0 && ind4 == face2->numberOfPoints() - 1) {
462  swap(ind1, ind2);
463  swap(ind3, ind4);
464  }
465  if (ind1 == 0 && ind2 == face1->numberOfPoints() - 1)
466  swap(ind1, ind2);
467  if (ind3 == 0 && ind4 == face2->numberOfPoints() - 1)
468  swap(ind3, ind4);
469  // remove all references to face1
470  for (size_t i=0; i<face1->numberOfPoints(); ++i)
471  face1->getPoint(i)->deleteFace(face1);
472  // remove all references to face2
473  for (size_t i=0; i<face2->numberOfPoints(); ++i)
474  face2->getPoint(i)->deleteFace(face2);
475  // add the new face
476  SubdivisionControlFace* newface = SubdivisionControlFace::construct(_owner);
477  newface->setLayer(layer);
478  _owner->addControlFace(newface);
479  for (size_t i=0; i<=ind1; ++i) {
480  newface->addPoint(face1->getPoint(i));
481  }
482  for (size_t i=ind4; i<face2->numberOfPoints(); ++i)
483  newface->addPoint(face2->getPoint(i));
484  for (size_t i=0; i<=ind3; ++i)
485  newface->addPoint(face2->getPoint(i));
486  for (size_t i=ind2; i<face1->numberOfPoints(); ++i)
487  newface->addPoint(face1->getPoint(i));
488  // check all appropriate points are added
489  if (newface->numberOfPoints() != face1->numberOfPoints() + face2->numberOfPoints() - 2)
490  throw runtime_error("wrong number of points in SubdivisionControlEdge::collapse");
491  p1 = newface->getPoint(newface->numberOfPoints() - 1);
492  for (size_t i=0; i<newface->numberOfPoints(); ++i) {
493  p2 = newface->getPoint(i);
494  SubdivisionEdge* edge = _owner->edgeExists(p1, p2);
495  if (edge != 0) {
496  if (edge->hasFace(face1))
497  edge->deleteFace(face1);
498  if (edge->hasFace(face2))
499  edge->deleteFace(face2);
500  edge->addFace(newface);
501  if (edge->numberOfFaces() < 2)
502  edge->setCrease(true);
503  }
504  p1 = p2;
505  }
506  // connect the new face to a layer
507  layer->addControlFace(newface);
508  if (isCrease())
509  setCrease(false);
510  startPoint()->deleteEdge(this);
511  endPoint()->deleteEdge(this);
512  if (_owner->hasControlEdge(this))
513  _owner->removeControlEdge(this);
514  if (_owner->hasControlFace(face1))
515  _owner->removeControlFace(face1);
516  if (_owner->hasControlFace(face2))
517  _owner->removeControlFace(face2);
518  _owner->deleteControlFace(face1);
519  _owner->deleteControlFace(face2);
520 
521  // check if startpoint and endpoint can be collapsed as well
522  if (s->numberOfFaces() > 1 && s->numberOfEdges() == 2)
523  s->collapse();
524  if (e->numberOfFaces() > 1 && e->numberOfEdges() == 2)
525  e->collapse();
526  _owner->setBuild(false);
527  // BUGBUG: pascal deletes edge here, but we can't do that...
528  } // end edge number == 2
529 }

606 SubdivisionControlPoint* SubdivisionControlEdge::insertControlPoint(const QVector3D& p)
607 {
608  SubdivisionControlPoint* result = SubdivisionControlPoint::construct(_owner);
609  SubdivisionControlPoint* sp = dynamic_cast<SubdivisionControlPoint*>(startPoint());
610  SubdivisionControlPoint* ep = dynamic_cast<SubdivisionControlPoint*>(endPoint());
611  if (sp == 0 || ep == 0)
612  throw runtime_error("start and/or end point are not control points");
613  result->setCoordinate(p);
614  if (getCurve() != 0) {
615  // insert the new point in the controlcurve
616  getCurve()->insertControlPoint(sp, ep, result);
617  }
618  _owner->addControlPoint(result);
619  for (size_t i=0; i<_faces.size(); ++i) {
620  SubdivisionFace* face = _faces[i];
621  if (face->hasPoint(sp) && face->hasPoint(ep)) {
622  size_t i1 = face->indexOfPoint(sp);
623  size_t i2 = face->indexOfPoint(ep);
624  if (i2 == i1+1)
625  face->insertPoint(i2, result);
626  else if (i1 == i2+1)
627  face->insertPoint(i1, result);
628  else if (i1 == 0 && i2 == face->numberOfPoints() - 1)
629  face->insertPoint(0, result);
630  else if (i2 == 0 && i1 == face->numberOfPoints() - 1)
631  face->insertPoint(0, result);
632  result->addFace(face);
633  }
634  }
635  ep->deleteEdge(this);
636  SubdivisionControlEdge* edge = _owner->addControlEdge(result, ep);
637  edge->setCrease(isCrease());
638  edge->setCurve(getCurve());
639  if (getCurve())
640  result->setVertexType(SubdivisionPoint::svCrease);
641  for (size_t i=0; i<_faces.size(); ++i)
642  edge->addFace(_faces[i]);
643  _points[1] = result;
644  result->addEdge(this);
645  return result;
646 }

704 void SubdivisionControlEdge::priv_trace(SubdivisionControlPoint* p)
705 {
706  SubdivisionControlEdge* edge;
707  if (p->isRegularPoint() && p->getVertexType() != SubdivisionPoint::svCorner) {
708  // find next edge
709  for (size_t i=0; i<p->numberOfEdges(); ++i) {
710  if (p->getEdge(i) != this) {
711  edge = dynamic_cast<SubdivisionControlEdge*>(p->getEdge(i));
712  if (edge->isSelected() != isSelected() && edge->isCrease() == isCrease()) {
713  bool shares_face = false;
714  for (size_t j=0; j<numberOfFaces(); ++j) {
715  if (edge->hasFace(_faces[j])) {
716  shares_face = true;
717  break;
718  }
719  }
720  if (!shares_face) {
721  if (edge->startPoint() == startPoint())
722  edge->swapData();
723  edge->setSelected(isSelected());
724  edge->trace();
725  break;
726  }
727  }
728  }
729  }

743 void SubdivisionControlEdge::draw(Viewport& vp, LineShader* lineshader)
744 {
745  if (!isVisible())
746  return;
747  QVector3D p1 = startPoint()->getCoordinate();
748  QVector3D p2 = endPoint()->getCoordinate();
749  QVector<QVector3D> vertices;
750  if (vp.getViewportMode() == Viewport::vmWireFrame) {
751  if (isCrease()) {
752  glLineWidth(2);
753  }
754  else {
755  glLineWidth(1);
756  }
757  if (vp.getViewportType() == Viewport::fvBodyplan) {
758  float mfl = _owner->getMainframeLocation();
759  if ((p1.x() < mfl && p2.x() > mfl) || (p1.x() > mfl && p2.x() < mfl)) {
760  // straddles mainframe
761  QVector3D m;
762  if (p2.x() - p1.x() != 0)
763  m = Interpolate(p1, p2, (mfl - p1.x()) / (p2.x() - p1.x()));
764  else
765  m = MidPoint(p1, p2);
766  if (p1.x() <= mfl) {
767  // p2 lies on port
768  vertices << p2;
769  vertices << m;
770  // p1 lies on stbd
771  p1.setY(-p1.y());
772  m.setY(-m.y());
773  vertices << p1;
774  vertices << m;
775  }
776  else {
777  // p2 lies on port
778  vertices << p1;
779  vertices << m;
780  // p2 lies on stbd
781  p2.setY(-p2.y());
782  m.setY(-m.y());
783  vertices << p2;
784  vertices << m;
785  }
786  }
787  else {
788  if (p1.x() <= mfl)
789  p1.setY(-p1.y());
790  if (p2.x() <= mfl)
791  p2.setY(-p2.y());
792  vertices << p1;
793  vertices << p2;
794  }
795  }
796  else {
797  vertices << p1;
798  vertices << p2;
799  }
800  }
801  else {
802  // viewport is not vmWireFrame
803  glLineWidth(1);
804  vertices << p1;
805  vertices << p2;
806  }
807  lineshader->renderLines(vertices, getColor());
808 }
