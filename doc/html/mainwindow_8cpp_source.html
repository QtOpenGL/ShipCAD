144 MainWindow::setAnimating(bool animating)
145 {
146  if (animating && _animation_timer != 0)
147  return;
148  if (!animating && _animation_timer != 0) {
149  _animation_timer->stop();

158  connect(_animation_timer, SIGNAL(timeout()), this, SLOT(animationTimeout()));
159  _animation_timer->start(100);
160  _vp->setAnimating(true);
161 }

163 void MainWindow::animationTimeout()
164 {
165  static float angle = 0;
166  static float elevation = 0;
167  static bool up = true;
168  angle += 5;
169  if (angle >= 180)
170  angle -= 360;
171  if (up)
172  elevation += 1;
173  else
174  elevation -= 1;
175  if (elevation >= 90) {
176  up = false;
177  elevation = 89;
178  }
179  if (elevation <= 0) {
180  up = true;
181  elevation = 0;
182  }
183  _vp->setAngle(angle);
184  _vp->setElevation(elevation);
185 }

187 void
188 MainWindow::openFile()
189 {
190  QString fileName = QFileDialog::getOpenFileName(this, tr("Open File"), QString(),
191  tr("All Files (*.*)"));
192 
193  if (!fileName.isEmpty()) {
194  QFile file(fileName);
195  if (!file.open(QIODevice::ReadOnly)) {
196  QMessageBox::critical(this, tr("Error"), tr("Could not open file"));
197  return;
198  }
199  QTextStream in(&file);
200  // read in file line by line
201  vector<QString> lines;
202  do {
203  QString line = in.readLine();
204  if (line.isNull())
205  break;
206  lines.push_back(line);
207  }
208  while (!in.atEnd());
209  file.close();
210  // now we have the file as lines, make a surface
211  SubdivisionSurface* newsurface = new SubdivisionSurface;
212  size_t lineno = 0;
213  newsurface->loadFromStream(lineno, lines);
214  // now set the gui to this new surface
215  SubdivisionSurface* old = _vp->getSurface();
216  if (old != 0) {
217  delete old;
218  }
219  setSurface(newsurface);
220  newsurface->setDesiredSubdivisionLevel(3);
221  newsurface->rebuild();
222  _vp->renderLater();
223  }
224 }
