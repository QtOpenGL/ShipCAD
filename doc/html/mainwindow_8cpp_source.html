186 void MainWindow::animationTimeout()
187 {
188  static float angle = 0;
189  static float elevation = 0;
190  static bool up = true;
191  angle += 5;
192  if (angle >= 180)
193  angle -= 360;
194  if (up)
195  elevation += 1;
196  else
197  elevation -= 1;
198  if (elevation >= 90) {
199  up = false;
200  elevation = 89;
201  }
202  if (elevation <= 0) {
203  up = true;
204  elevation = 0;
205  }
206  _vp->setAngle(angle);
207  _vp->setElevation(elevation);
208 }

210 void
211 MainWindow::openFile()
212 {
213  QString fileName = QFileDialog::getOpenFileName(this, tr("Open File"), QString(),
214  tr("All Files (*.*)"));
215 
216  if (!fileName.isEmpty()) {
217  QFile file(fileName);
218  if (!file.open(QIODevice::ReadOnly)) {
219  QMessageBox::critical(this, tr("Error"), tr("Could not open file"));
220  return;
221  }
222  QTextStream in(&file);
223  // read in file line by line
224  QStringList lines;
225  do {
226  QString line = in.readLine();
227  if (line.isNull())
228  break;
229  lines.push_back(line);
230  }
231  while (!in.atEnd());
232  file.close();
233  // now we have the file as lines, make a surface
234  SubdivisionSurface* newsurface = new SubdivisionSurface;
235  size_t lineno = 0;
236  newsurface->loadFromStream(lineno, lines);
237  // now set the gui to this new surface
238  SubdivisionSurface* old = _vp->getSurface();
239  if (old != 0) {
240  delete old;
241  }
242  setSurface(newsurface);
243  newsurface->setDesiredSubdivisionLevel(3);
244  newsurface->rebuild();
245  _vp->renderLater();
246  }
