93 bool Controller::shootPickRay(Viewport& vp, const PickRay& ray)
94 {
95  bool scene_changed = false;
96  vector<SubdivisionBase*> filtered;
97  multimap<float, SubdivisionBase*> picks = getModel()->getSurface()->shootPickRay(vp, ray);
98  if (picks.size() > 0) {
99  multimap<float, SubdivisionBase*>::iterator it, itlow;
100  itlow = picks.lower_bound(0);
101  float last = 0;
102  bool first = true;
103  for (it=itlow; it!=picks.end(); ++it) {
104  float s = (*it).first;
105  // check to see if next element is farther away than first, since this is multimap, we
106  // can several at the same distance from the ray origin
107  if (!first) {
108  if (s > last)
109  break;
110  } else
111  first = false;
112  filtered.push_back((*it).second);
113  last = s;
114  }
115  }
116  if (filtered.size() > 0) {
117  if (filtered.size() == 1) {
118  // is this a point?
119  SubdivisionControlPoint* cp = dynamic_cast<SubdivisionControlPoint*>(filtered[0]);
120  if (cp != 0) {
121  cp->setSelected(true);
122  scene_changed = true;
123  cout << "control point selected" << endl;
124  }
125  } else {
126  // need to discriminate against these elements
127  throw runtime_error("multiple elements picked");
128  }
129  }
130  if (scene_changed) {
131  // TODO file changed, undo
132  getModel()->setFileChanged(true);
133  emit changeSelectedItems();
134  emit modelGeometryChanged();
135  }
136  return scene_changed;
137 }

149 UndoObject* Controller::createRedoObject()
150 {
151  version_t version = _model->getFileVersion();
152  bool preview = _model->getProjectSettings().getSavePreview();
153  // temporarily set to the latest fileversion so no data will be lost
154  UndoObject* rd = new UndoObject(_model,
155  _model->getFilename(),
156  _model->getEditMode(),
157  _model->isFileChanged(),
158  _model->isFilenameSet(),
159  true);
160  try {
161  _model->setFileVersion(k_current_version);
162  _model->getProjectSettings().setSavePreview(false);
163  _model->saveBinary(rd->getUndoData());
164  rd->accept();
165  emit updateUndoData();
166  } catch(...) {
167  // restore the original version
168  _model->setFileVersion(version);
169  _model->getProjectSettings().setSavePreview(preview);
170  delete rd;
171  rd = 0;
172  }
173  return rd;
174 }

802 void Controller::cornerPointSelected(bool sel)
803 {
804  cout << "corner point selected:" << (sel ? 'y' : 'n') << endl;
805  if (_model->getActiveControlPoint() != 0) {
806  SubdivisionControlPoint* ap = _model->getActiveControlPoint();
807  // count the number of crease edges connected to this point
808  vertex_type_t oldtype = ap->getVertexType();
809  if (ap->getVertexType() == svCorner) {
810  size_t n = 0;
811  for (size_t i=0; i<ap->numberOfEdges(); i++) {
812  if (ap->getEdge(i)->isCrease())
813  n++;
814  }
815  // count the number of incident crease edges
816  if (n == 0)
817  ap->setVertexType(svRegular);
818  else if (n == 1)
819  ap->setVertexType(svDart);
820  else if (n == 2)
821  ap->setVertexType(svCrease);
822  } else
823  ap->setVertexType(svCorner);
824  if (ap->getVertexType() != oldtype) {
825  // create an undo object
826  emit updateControlPointValue(ap);
827  emit modelGeometryChanged();
828  }
829  }
